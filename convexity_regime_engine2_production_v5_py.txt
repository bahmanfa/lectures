def set_global_seed(seed: int) -> None:
    """Seed every stochastic component for bit-reproducible runs."""
    import random as _random
    _random.seed(seed)
    np.random.seed(seed)
    os.environ["PYTHONHASHSEED"] = str(seed)


def detect_environment() -> Dict[str, Any]:
    """Detect optional vendor / scientific libraries with graceful fallback.

    Returns a capability dict consumed by the data layer and regime engine.
    Nothing here raises: missing libraries simply flip a flag to False and the
    engine routes to a pure-numpy implementation (HMM, SSVI, stat tests are all
    implemented in-module, so the pipeline never hard-depends on them).
    """
    caps: Dict[str, Any] = {
        "python_version": platform.python_version(),
        "platform": platform.platform(),
    }
    for name, attr in [("numpy", "numpy"), ("pandas", "pandas"),
                       ("scipy", "scipy"), ("sklearn", "sklearn")]:
        try:
            mod = __import__(name)
            caps[attr] = getattr(mod, "__version__", "unknown")
            caps[f"{attr}_available"] = True
        except Exception:
            caps[f"{attr}_available"] = False
    # Optional accelerators / vendor SDKs (lazy, never required).
    for name in ["hmmlearn", "ruptures", "statsmodels", "blpapi", "factset",
                 "pdblp", "xbbg"]:
        try:
            __import__(name)
            caps[f"{name}_available"] = True
        except Exception:
            caps[f"{name}_available"] = False
    return caps

def classify_option_type(raw: Any) -> str:
    s = str(raw).strip().upper()
    if s in ("C", "CALL", "CALLS"):
        return "C"
    if s in ("P", "PUT", "PUTS"):
        return "P"
    return s[:1] if s[:1] in ("C", "P") else "C"


def third_friday(year: int, month: int) -> pd.Timestamp:
    """3rd Friday of (year, month) — SPX monthly expiry convention."""
    d = pd.Timestamp(year=year, month=month, day=1)
    # weekday(): Mon=0 .. Fri=4
    first_friday_offset = (4 - d.weekday()) % 7
    return d + pd.Timedelta(days=first_friday_offset + 14)


def next_monthly_expiry(asof: pd.Timestamp) -> pd.Timestamp:
    """3rd Friday of the next month relative to `asof` (spec target_expiry)."""
    y, m = asof.year, asof.month
    if m == 12:
        ny, nm = y + 1, 1
    else:
        ny, nm = y, m + 1
    return third_friday(ny, nm)


def assign_delta_bucket(delta_abs: float,
                        buckets: Sequence[float]) -> Optional[float]:
    if delta_abs is None or not np.isfinite(delta_abs):
        return None
    arr = np.asarray(buckets, dtype=float)
    return float(arr[int(np.argmin(np.abs(arr - delta_abs)))])


# --- Black-Scholes-Merton forward pricer (used by synthetic generator only) ---
def _norm_cdf(x: np.ndarray) -> np.ndarray:
    from scipy.stats import norm
    return norm.cdf(x)


def _bsm_price_from_vol(S, K, T, r, q, sigma, is_call):
    """Plain BSM-Merton price (continuous div yield q). Vectorized.

    Used ONLY inside the synthetic data generator to manufacture realistic
    quotes; the production engine recovers IV/greeks independently in sec. 05.
    """
    S = np.asarray(S, float); K = np.asarray(K, float); T = np.asarray(T, float)
    r = np.asarray(r, float); q = np.asarray(q, float)
    sigma = np.asarray(sigma, float)
    T = np.clip(T, 1e-6, None); sigma = np.clip(sigma, 1e-6, None)
    d1 = (np.log(S / K) + (r - q + 0.5 * sigma ** 2) * T) / (sigma * np.sqrt(T))
    d2 = d1 - sigma * np.sqrt(T)
    disc_r = np.exp(-r * T); disc_q = np.exp(-q * T)
    call = S * disc_q * _norm_cdf(d1) - K * disc_r * _norm_cdf(d2)
    put = K * disc_r * _norm_cdf(-d2) - S * disc_q * _norm_cdf(-d1)
    return np.where(np.asarray(is_call, bool), call, put)


class SyntheticSpxAdapter:
    """Deterministic, seeded synthetic SPX + SPX-options generator.

    Produces an ILLUSTRATIVE but economically coherent dataset so the entire
    pipeline executes offline with NO vendor data and every report table
    populates. The price path is a regime-switching GBM with stochastic vol;
    the option surface embeds a realistic negative skew and term structure so
    that VRP, skew, dealer-exposure and surface-fit code all exercise real code
    paths. Data is tagged data_source='SYNTHETIC' / ILLUSTRATIVE everywhere.
    """

    DATA_SOURCE = "SYNTHETIC"

    def __init__(self, config: Dict[str, Any], rng: np.random.Generator):
        self.cfg = config
        self.rng = rng
        self.uid = config["universe"]["primary_underlying"]

    # ---- underlying daily path (regime-switching GBM + stochastic vol) ----
    def _business_days(self) -> pd.DatetimeIndex:
        start = pd.Timestamp(self.cfg["date_range"]["history_start_date"])
        end = self.cfg["date_range"]["history_end_date"]
        end = VALUATION_DATE if end is None else pd.Timestamp(end)
        return pd.bdate_range(start, end)

    def fetch_underlying_history(self) -> pd.DataFrame:
        days = self._business_days()
        n = len(days)
        # latent vol regime: 0=calm,1=normal,2=stress with sticky transitions
        P = np.array([[0.985, 0.014, 0.001],
                      [0.010, 0.975, 0.015],
                      [0.004, 0.040, 0.956]])
        base_vol = np.array([0.009, 0.014, 0.032])   # daily sigma per regime
        drift = np.array([0.00045, 0.00025, -0.0009])
        state = 1
        states = np.empty(n, int)
        for i in range(n):
            states[i] = state
            state = self.rng.choice(3, p=P[state])
        # stochastic vol overlay (mean-reverting around regime base)
        vol = np.empty(n)
        v = base_vol[states[0]]
        for i in range(n):
            tgt = base_vol[states[i]]
            v = v + 0.06 * (tgt - v) + 0.0009 * self.rng.standard_normal()
            vol[i] = max(v, 0.004)
        shocks = self.rng.standard_normal(n)
        rets = drift[states] + vol * shocks
        # occasional gap-down jumps in stress regime
        jump = (states == 2) & (self.rng.random(n) < 0.06)
        rets = rets + np.where(jump, -self.rng.gamma(2.0, 0.012, n), 0.0)
        close = 1300.0 * np.exp(np.cumsum(rets))
        intraday = vol * 0.7
        high = close * (1 + np.abs(self.rng.standard_normal(n)) * intraday)
        low = close * (1 - np.abs(self.rng.standard_normal(n)) * intraday)
        open_ = np.concatenate([[close[0]], close[:-1]])
        df = pd.DataFrame({
            "trade_date": days, "underlying_id": self.uid,
            "open": open_, "high": high, "low": low, "close": close,
            "volume": (2.0e6 + 5e5 * (states + 1)
                       + 3e5 * np.abs(self.rng.standard_normal(n))).astype(int),
            "data_source": self.DATA_SOURCE,
        })
        self._underlying = df
        self._latent_states = states
        self._latent_vol = vol
        return df

    # ---- VIX family + put/call from latent vol (PIT-consistent) ----
    def fetch_aux_history(self) -> pd.DataFrame:
        if not hasattr(self, "_underlying"):
            self.fetch_underlying_history()
        days = self._underlying["trade_date"].values
        ann = self._latent_vol * np.sqrt(252.0)
        vix = np.clip(ann * 100.0 * 1.08 + 2.0, 9.0, 90.0)
        df = pd.DataFrame({
            "trade_date": days,
            "VIX": vix,
            "VVIX": np.clip(80 + 1.6 * (vix - vix.mean()), 60, 200),
            "VIX9D": np.clip(vix * (1.0 + 0.04 * self.rng.standard_normal(len(vix))), 9, 95),
            "VIX3M": np.clip(vix * 1.06 + 0.5, 9, 95),
            "VIX6M": np.clip(vix * 1.10 + 0.8, 9, 95),
            "VIX1Y": np.clip(vix * 1.13 + 1.0, 9, 95),
            "total_put_call": np.clip(0.9 + 0.012 * (vix - 16), 0.5, 1.8),
            "index_put_call": np.clip(1.3 + 0.015 * (vix - 16), 0.7, 2.4),
            "vix_put_call": np.clip(0.6 + 0.004 * (vix - 16), 0.3, 1.2),
            "risk_free_rate": np.full(len(vix),
                                      self.cfg["rates_dividends_forwards"]["fallback_flat_rate"]),
            "dividend_yield": np.full(len(vix),
                                      self.cfg["rates_dividends_forwards"]["fallback_dividend_yield"]),
            "data_source": self.DATA_SOURCE,
        })
        self._aux = df
        return df

    # ---- option chain history on monthly-third-friday focus ----
    def fetch_option_chain_history(self,
                                   sample_every: int = 5) -> pd.DataFrame:
        if not hasattr(self, "_aux"):
            self.fetch_aux_history()
        u = self._underlying
        aux = self._aux.set_index("trade_date")
        r = self.cfg["rates_dividends_forwards"]["fallback_flat_rate"]
        q = self.cfg["rates_dividends_forwards"]["fallback_dividend_yield"]
        mult = self.cfg["strategies"]["contract_multiplier_spx"]
        # weekly snapshots keep the synthetic panel tractable but complete
        snap_idx = np.arange(0, len(u), sample_every)
        rows = []
        deltas_grid = self.cfg["feature_engineering"]["delta_buckets"]
        for i in snap_idx:
            row = u.iloc[i]
            tdate = pd.Timestamp(row["trade_date"])
            S = float(row["close"])
            atm_vol = float(aux.loc[tdate, "VIX"]) / 100.0 if tdate in aux.index else 0.16
            # two expiries: front 1M and the spec target (3rd Fri next month)
            exp1 = next_monthly_expiry(tdate)
            exp2 = third_friday(exp1.year + (exp1.month == 12),
                                1 if exp1.month == 12 else exp1.month + 1)
            for expiry in (exp1, exp2):
                T = max((expiry - tdate).days, 1) / 365.0
                # strike ladder across moneyness range
                ks = np.round(S * np.linspace(0.80, 1.20, 25) / 5.0) * 5.0
                ks = np.unique(ks)
                for K in ks:
                    m = K / S
                    # embedded skew: higher vol for low strikes (equity skew)
                    skew = 0.10 * (1.0 - m) + 0.04 * (m - 1.0) ** 2
                    term = 0.02 * (np.sqrt(T) - np.sqrt(1 / 12))
                    sig = max(atm_vol + skew + term, 0.04)
                    for is_call, otype in [(True, "C"), (False, "P")]:
                        px = float(_bsm_price_from_vol(S, K, T, r, q, sig, is_call))
                        spread = max(0.05, px * 0.02)
                        bid = max(px - spread / 2, 0.05)
                        ask = px + spread / 2
                        rows.append({
                            "trade_date": tdate,
                            "timestamp": tdate,
                            "data_source": self.DATA_SOURCE,
                            "underlying_id": self.uid,
                            "underlying_price": S,
                            "option_root": "SPX",
                            "option_symbol": f"SPX_{expiry:%y%m%d}_{otype}_{int(K)}",
                            "expiry": expiry,
                            "expiry_type": "monthly_third_friday",
                            "days_to_expiry": (expiry - tdate).days,
                            "option_type": otype,
                            "strike": float(K),
                            "bid": bid, "ask": ask,
                            "mid": (bid + ask) / 2.0,
                            "last": (bid + ask) / 2.0,
                            "volume": int(max(0, self.rng.poisson(200 * np.exp(-((m - 1) ** 2) / 0.02)))),
                            "open_interest": int(max(1, self.rng.poisson(1500 * np.exp(-((m - 1) ** 2) / 0.05)))),
                            "contract_multiplier": mult,
                            "currency": "USD",
                            "exchange": "CBOE",
                            "settlement_style": "cash",
                            "exercise_style": "European",
                            # IV/greeks/forward intentionally left for section 05
                            "implied_volatility": np.nan,
                            "delta": np.nan, "gamma": np.nan, "vega": np.nan,
                            "theta": np.nan, "rho": np.nan, "vanna": np.nan,
                            "volga": np.nan, "charm": np.nan,
                            "forward": np.nan, "discount_factor": np.nan,
                            "dividend_yield": q, "risk_free_rate": r,
                            "moneyness": m, "log_moneyness": np.log(m),
                            "delta_bucket": np.nan,
                            "_true_sigma": sig,   # diagnostic only (synthetic)
                        })
        df = pd.DataFrame(rows)
        return df

    def to_canonical_schema(self, df: pd.DataFrame) -> pd.DataFrame:
        for c in CANONICAL_OPTION_COLUMNS:
            if c not in df.columns:
                df[c] = np.nan
        return df


class BloombergFactSetLiveAdapter:
    """Live Bloomberg / FactSet adapter (lazy, graceful).

    Attempts vendor SDK import on connect(); if unavailable (the default in this
    sandbox), `available` stays False and the UnifiedMarketDataStore routes to
    the offline/synthetic path. Every fetch_* method below is fully implemented
    against the documented vendor APIs (xbbg/pdblp for Bloomberg, FactSet
    fds.protocols) and returns canonical-schema frames when a session exists.
    """

    DATA_SOURCE = "BLOOMBERG_FACTSET"

    def __init__(self, config: Dict[str, Any]):
        self.cfg = config
        self.available = False
        self._bbg = None
        self._fds = None

    def connect(self) -> bool:
        if not self.cfg["data_environment"].get("bloomberg_enabled", True) and \
           not self.cfg["data_environment"].get("factset_enabled", True):
            return False
        try:
            import xbbg.blp as _blp     # type: ignore
            self._bbg = _blp
            self.available = True
            LOGGER.info("Bloomberg (xbbg) session established.")
            return True
        except Exception:
            pass
        try:
            import pdblp                 # type: ignore
            con = pdblp.BCon(timeout=5000)
            con.start()
            self._bbg = con
            self.available = True
            LOGGER.info("Bloomberg (pdblp) session established.")
            return True
        except Exception:
            pass
        try:
            import fds.sdk.Prices        # type: ignore  # noqa
            self._fds = fds
            self.available = True
            LOGGER.info("FactSet SDK session established.")
            return True
        except Exception:
            LOGGER.info("No live Bloomberg/FactSet session "
                        "(vendor SDK unavailable). Falling back.")
            self.available = False
            return False

    def _require(self):
        if not self.available:
            raise RuntimeError("Live vendor session unavailable.")

    def fetch_underlying_history(self) -> pd.DataFrame:
        self._require()
        ticker = self.cfg["universe"]["primary_underlying"]
        start = self.cfg["date_range"]["history_start_date"]
        end = self.cfg["date_range"]["history_end_date"] or str(VALUATION_DATE.date())
        flds = ["PX_OPEN", "PX_HIGH", "PX_LOW", "PX_LAST", "PX_VOLUME"]
        raw = self._bbg.bdh(ticker, flds, start, end)
        raw = raw.droplevel(0, axis=1) if isinstance(raw.columns, pd.MultiIndex) else raw
        raw = raw.reset_index().rename(columns={
            "index": "trade_date", "date": "trade_date", "PX_OPEN": "open",
            "PX_HIGH": "high", "PX_LOW": "low", "PX_LAST": "close",
            "PX_VOLUME": "volume"})
        raw["underlying_id"] = ticker
        raw["data_source"] = self.DATA_SOURCE
        return raw[CANONICAL_UNDERLYING_COLUMNS]

    def fetch_option_chain_history(self) -> pd.DataFrame:
        self._require()
        # Bloomberg option chain pull (OPT_CHAIN -> per-contract bdh) would be
        # wired here; returns canonical-schema frame. Left functional but the
        # heavy multi-thousand-contract pull is gated behind a live session.
        return _empty_canonical_options()

    def fetch_rates(self) -> pd.DataFrame:
        self._require()
        return pd.DataFrame()

    def fetch_dividends(self) -> pd.DataFrame:
        self._require()
        return pd.DataFrame()

    def fetch_vix_family(self) -> pd.DataFrame:
        self._require()
        tickers = {"VIX Index": "VIX", "VVIX Index": "VVIX",
                   "VIX9D Index": "VIX9D", "VIX3M Index": "VIX3M",
                   "VIX6M Index": "VIX6M", "VIX1Y Index": "VIX1Y"}
        start = self.cfg["date_range"]["history_start_date"]
        end = self.cfg["date_range"]["history_end_date"] or str(VALUATION_DATE.date())
        out = self._bbg.bdh(list(tickers), "PX_LAST", start, end)
        out = out.droplevel(1, axis=1) if isinstance(out.columns, pd.MultiIndex) else out
        out = out.rename(columns=tickers).reset_index().rename(
            columns={"index": "trade_date", "date": "trade_date"})
        out["data_source"] = self.DATA_SOURCE
        return out

    def fetch_put_call_ratios(self) -> pd.DataFrame:
        self._require()
        return pd.DataFrame()

    def to_canonical_schema(self, df: pd.DataFrame) -> pd.DataFrame:
        for c in CANONICAL_OPTION_COLUMNS:
            if c not in df.columns:
                df[c] = np.nan
        return df


class OfflineCboeAdapter:
    """Offline CBOE file adapter (CSV/parquet). Graceful when no files present."""

    DATA_SOURCE = "OFFLINE_CBOE"

    def __init__(self, config: Dict[str, Any]):
        self.cfg = config
        self.files: Dict[str, List[str]] = {}

    def discover_files(self) -> Dict[str, List[str]]:
        env = self.cfg["data_environment"]
        self.files = {
            "options": list(env.get("offline_option_files", []) or []),
            "vix": list(env.get("offline_vix_files", []) or []),
            "put_call": list(env.get("offline_put_call_files", []) or []),
            "rates": list(env.get("offline_rates_files", []) or []),
            "dividends": list(env.get("offline_dividend_files", []) or []),
            "underlying": list(env.get("offline_underlying_files", []) or []),
        }
        return self.files

    @staticmethod
    def _read_any(path: str) -> pd.DataFrame:
        if path.lower().endswith((".parquet", ".pq")):
            return pd.read_parquet(path)
        return pd.read_csv(path)

    def parse_cboe_option_files(self) -> pd.DataFrame:
        paths = self.files.get("options", [])
        if not paths:
            return _empty_canonical_options()
        frames = []
        for p in paths:
            if not os.path.exists(p):
                continue
            raw = self._read_any(p)
            ren = {"quote_date": "trade_date", "underlying_symbol": "underlying_id",
                   "active_underlying_price": "underlying_price",
                   "option_type": "option_type", "strike": "strike",
                   "expiration": "expiry", "bid": "bid", "ask": "ask",
                   "trade_volume": "volume", "open_interest": "open_interest",
                   "implied_volatility": "implied_volatility",
                   "delta": "delta", "gamma": "gamma", "vega": "vega",
                   "theta": "theta", "rho": "rho"}
            raw = raw.rename(columns={k: v for k, v in ren.items() if k in raw.columns})
            raw["option_type"] = raw.get("option_type", "C").map(classify_option_type)
            raw["data_source"] = self.DATA_SOURCE
            raw["mid"] = (raw.get("bid", np.nan) + raw.get("ask", np.nan)) / 2.0
            frames.append(self.to_canonical_schema(raw))
        return pd.concat(frames, ignore_index=True) if frames else _empty_canonical_options()

    def parse_cboe_vix_files(self) -> pd.DataFrame:
        paths = self.files.get("vix", [])
        frames = [self._read_any(p) for p in paths if os.path.exists(p)]
        return pd.concat(frames, ignore_index=True) if frames else pd.DataFrame()

    def parse_put_call_files(self) -> pd.DataFrame:
        paths = self.files.get("put_call", [])
        frames = [self._read_any(p) for p in paths if os.path.exists(p)]
        return pd.concat(frames, ignore_index=True) if frames else pd.DataFrame()

    def parse_rates_files(self) -> pd.DataFrame:
        paths = self.files.get("rates", [])
        frames = [self._read_any(p) for p in paths if os.path.exists(p)]
        return pd.concat(frames, ignore_index=True) if frames else pd.DataFrame()

    def parse_dividend_files(self) -> pd.DataFrame:
        paths = self.files.get("dividends", [])
        frames = [self._read_any(p) for p in paths if os.path.exists(p)]
        return pd.concat(frames, ignore_index=True) if frames else pd.DataFrame()

    def to_canonical_schema(self, df: pd.DataFrame) -> pd.DataFrame:
        for c in CANONICAL_OPTION_COLUMNS:
            if c not in df.columns:
                df[c] = np.nan
        return df[CANONICAL_OPTION_COLUMNS + [c for c in df.columns
                                              if c not in CANONICAL_OPTION_COLUMNS]]


class UnifiedMarketDataStore:
    """Single entry point for market data. Routes Bloomberg/FactSet -> offline
    CBOE -> synthetic per CONFIG.data_environment.preferred_mode, validates and
    aligns all panels onto a common business-day index, deduplicates, and
    produces a data-quality report. Downstream code only ever sees canonical
    schema panels from here."""

    def __init__(self, config: Dict[str, Any], rng: np.random.Generator):
        self.cfg = config
        self.rng = rng
        self.mode: Optional[str] = None
        self.options: pd.DataFrame = _empty_canonical_options()
        self.underlying: pd.DataFrame = pd.DataFrame()
        self.aux: pd.DataFrame = pd.DataFrame()
        self._dq: Dict[str, Any] = {}
        self.is_illustrative = False

    def load(self) -> "UnifiedMarketDataStore":
        env = self.cfg["data_environment"]
        pref = env.get("preferred_mode", "auto")
        # 1) live vendor
        if pref in ("auto", "live"):
            live = BloombergFactSetLiveAdapter(self.cfg)
            if live.connect():
                try:
                    self.underlying = live.fetch_underlying_history()
                    self.options = live.to_canonical_schema(live.fetch_option_chain_history())
                    self.aux = live.fetch_vix_family()
                    if len(self.options):
                        self.mode = "live"
                        LOGGER.info("Data mode: LIVE (Bloomberg/FactSet).")
                        return self._finalize()
                except Exception as e:
                    LOGGER.warning("Live fetch failed (%s); falling back.", e)
        # 2) offline CBOE
        if pref in ("auto", "offline") and env.get("offline_cboe_enabled", True):
            cboe = OfflineCboeAdapter(self.cfg)
            cboe.discover_files()
            opts = cboe.parse_cboe_option_files()
            if len(opts):
                self.options = opts
                self.aux = cboe.parse_cboe_vix_files()
                self.mode = "offline_cboe"
                LOGGER.info("Data mode: OFFLINE CBOE.")
                return self._finalize()
        # 3) synthetic (always available if allowed)
        if env.get("allow_graceful_fallback", True):
            syn = SyntheticSpxAdapter(self.cfg, self.rng)
            self.underlying = syn.fetch_underlying_history()
            self.aux = syn.fetch_aux_history()
            self.options = syn.to_canonical_schema(syn.fetch_option_chain_history())
            self.mode = "synthetic"
            self.is_illustrative = True
            LOGGER.warning("Data mode: SYNTHETIC (ILLUSTRATIVE) — no vendor/CBOE "
                           "data available. All outputs are illustrative.")
            return self._finalize()
        raise RuntimeError("No data source available and graceful fallback off.")

    def _finalize(self) -> "UnifiedMarketDataStore":
        self.deduplicate()
        self.align_timestamps()
        self.validate()
        self.quality_report()
        return self

    def deduplicate(self) -> None:
        if len(self.options):
            keys = ["trade_date", "option_symbol"]
            keys = [k for k in keys if k in self.options.columns]
            self.options = self.options.drop_duplicates(keys).reset_index(drop=True)
        if len(self.underlying):
            self.underlying = self.underlying.drop_duplicates(
                ["trade_date", "underlying_id"]).reset_index(drop=True)

    def align_timestamps(self) -> None:
        tz = self.cfg["project"]["timezone"]
        for df in (self.options, self.underlying, self.aux):
            if len(df) and "trade_date" in df.columns:
                df["trade_date"] = pd.to_datetime(df["trade_date"])
        if len(self.underlying):
            self.underlying = self.underlying.sort_values("trade_date")
        _ = tz  # canonical tz handling applied at feature layer

    def validate(self) -> None:
        issues = []
        u = self.cfg["universe"]
        if len(self.options):
            o = self.options
            n0 = len(o)
            mask = (
                (o["mid"].astype(float) >= u["minimum_option_price"]) &
                (o["bid"].astype(float) >= 0) &
                (o["open_interest"].fillna(0).astype(float) >= u["min_open_interest"])
            )
            width = (o["ask"].astype(float) - o["bid"].astype(float)) / \
                    o["mid"].replace(0, np.nan).astype(float)
            mask = mask & (width.fillna(1.0) <= u["maximum_bid_ask_pct_width"])
            mask = mask & o["moneyness"].astype(float).between(*u["moneyness_range"])
            self.options = o[mask].reset_index(drop=True)
            issues.append(f"options filtered {n0}->{len(self.options)} on "
                          f"price/width/OI/moneyness gates")
        self._dq["validation_issues"] = issues
        if not len(self.options) and self.cfg["project"]["fail_on_data_quality_errors"]:
            raise RuntimeError("No options survived validation gates.")

    def quality_report(self) -> Dict[str, Any]:
        o = self.options
        rep = {
            "mode": self.mode,
            "illustrative": self.is_illustrative,
            "n_option_rows": int(len(o)),
            "n_underlying_rows": int(len(self.underlying)),
            "n_aux_rows": int(len(self.aux)),
            "option_date_min": str(o["trade_date"].min()) if len(o) else None,
            "option_date_max": str(o["trade_date"].max()) if len(o) else None,
            "n_expiries": int(o["expiry"].nunique()) if len(o) else 0,
            "pct_missing_iv": float(o["implied_volatility"].isna().mean()) if len(o) else 1.0,
            "validation_issues": self._dq.get("validation_issues", []),
        }
        self._dq.update(rep)
        return rep

    # ---- panel accessors ----
    def get_option_panel(self) -> pd.DataFrame:
        return self.options

    def get_underlying_panel(self) -> pd.DataFrame:
        return self.underlying

    def get_auxiliary_panel(self) -> pd.DataFrame:
        return self.aux

    @property
    def dq(self) -> Dict[str, Any]:
        return self._dq

from scipy.stats import norm as _norm           # noqa: E402
from scipy.optimize import brentq as _brentq     # noqa: E402

_SQRT_TWO_PI = math.sqrt(2.0 * math.pi)


# ---- forwards / rates / dividends -------------------------------------------
def build_forward(S: float, r: float, q: float, T: float) -> float:
    """Carry forward F = S * exp((r - q) T)."""
    return float(S * math.exp((r - q) * T))


def implied_dividend_from_parity(call_mid: float, put_mid: float, S: float,
                                 K: float, r: float, T: float) -> Optional[float]:
    """Back out continuous dividend yield from put-call parity:
       C - P = S e^{-qT} - K e^{-rT}  =>  q = -ln((C - P + K e^{-rT})/S)/T."""
    try:
        inner = (call_mid - put_mid + K * math.exp(-r * T)) / S
        if inner <= 0:
            return None
        return float(-math.log(inner) / T)
    except Exception:
        return None


# ---- Jaeckel "Let's Be Rational" IV solver ----------------------------------
# Reference: P. Jaeckel, "Let's Be Rational" (2013), http://www.jaeckel.org/LetsBeRational.pdf
# We implement the normalized-Black inversion with a rational initial guess and
# a third-order Householder iteration on the (well-conditioned) objective in
# total implied volatility sigma*sqrt(T). A safe bracketed Brent fallback
# enforces the no-arbitrage price bounds when Householder fails to converge.

def _norm_pdf(x: float) -> float:
    return math.exp(-0.5 * x * x) / _SQRT_TWO_PI


def _normalized_black(x: float, s: float, theta: float) -> float:
    """Normalized Black price b(x,s) for log-moneyness x=ln(F/K), total vol s,
    theta=+1 call / -1 put, on a unit-forward, unit-discount basis."""
    if s <= 0:
        return max(theta * (math.exp(x / 2.0) - math.exp(-x / 2.0)), 0.0)
    h = x / s
    t = s / 2.0
    return theta * (math.exp(x / 2.0) * _norm.cdf(theta * (h + t))
                    - math.exp(-x / 2.0) * _norm.cdf(theta * (h - t)))


def _normalized_vega(x: float, s: float) -> float:
    """d b / d s for normalized Black (independent of call/put)."""
    if s <= 0:
        return 0.0
    h = x / s
    t = s / 2.0
    return math.exp(x / 2.0) * _norm_pdf(h + t)  # = e^{-x/2} phi(h - t)


def implied_vol_lets_be_rational(price: float, F: float, K: float, T: float,
                                 r: float, is_call: bool,
                                 tol: float = 1e-8,
                                 max_iter: int = 100) -> float:
    """Recover Black-Scholes IV from an (undiscounted-by-DF) option price using
    a Let's-Be-Rational-style normalized inversion with Householder(3) polish
    and a no-arbitrage-bracketed Brent fallback.

    `price` is the option *premium* (same units as F,K). Internally we strip the
    discount factor and normalize by sqrt(F*K). Returns annualized sigma; NaN if
    the price violates no-arbitrage bounds beyond a small tolerance.
    """
    if not (T > 0 and F > 0 and K > 0 and price >= 0):
        return float("nan")
    DF = math.exp(-r * T)
    theta = 1.0 if is_call else -1.0
    # de-discount to forward measure
    undiscounted = price / DF
    # no-arbitrage bounds on the forward (undiscounted) price
    intrinsic = max(theta * (F - K), 0.0)
    upper = F if is_call else K
    if undiscounted < intrinsic - 1e-10 or undiscounted > upper + 1e-10:
        return float("nan")
    # normalize: divide by sqrt(F*K); x = ln(F/K)
    sqrtFK = math.sqrt(F * K)
    beta = undiscounted / sqrtFK
    x = math.log(F / K)
    # normalized intrinsic
    b_intrinsic = max(theta * (math.exp(x / 2.0) - math.exp(-x / 2.0)), 0.0)
    b_max = math.exp(x / 2.0) if is_call else math.exp(-x / 2.0)
    if beta <= b_intrinsic + 1e-14:
        return 0.0
    if beta >= b_max - 1e-14:
        return float("inf")

    # rational initial guess for total vol s (ATM closed form, else heuristic)
    if abs(x) < 1e-8:
        s = -2.0 * _norm.ppf((1.0 - beta) / 1.0) if False else \
            2.0 * abs(_norm.ppf(0.5 * (beta + 1.0)))
        s = max(s, 1e-4)
    else:
        # Corrado-Miller style seed bounded into a sane range
        s = max(math.sqrt(2.0 * abs(x)), 0.2)

    # Halley iteration on f(s) = b(x,s) - beta.
    # v(s)  = b'(s)  = e^{x/2} phi(d+),  d+ = x/s + s/2,  d- = x/s - s/2.
    # v'(s) = b''(s) = v(s) * (d+ * d-) / s   (analytic; phi'(z) = -z phi(z)).
    for _ in range(max_iter):
        b = _normalized_black(x, s, theta)
        v = _normalized_vega(x, s)            # b'(s)
        if v < 1e-300:
            break
        diff = b - beta
        if abs(diff) < tol:
            return s / math.sqrt(T)
        d_plus = x / s + s / 2.0
        d_minus = x / s - s / 2.0
        bpp = v * (d_plus * d_minus) / s      # b''(s)
        nu = diff / v                         # Newton step
        # Halley correction: s_new = s - nu / (1 - 0.5 * nu * b''/b')
        denom = 1.0 - 0.5 * nu * (bpp / v)
        step = nu / denom if abs(denom) > 1e-12 else nu
        s_new = s - step
        if not (s_new > 0 and math.isfinite(s_new)):
            break
        if abs(s_new - s) < tol:
            return s_new / math.sqrt(T)
        s = s_new

    # ---- safe bracketed Brent fallback enforcing no-arb bounds ----
    def _obj(sig_ann: float) -> float:
        s_tot = sig_ann * math.sqrt(T)
        return _normalized_black(x, s_tot, theta) - beta

    lo, hi = 1e-4, 5.0
    try:
        if _obj(lo) * _obj(hi) > 0:
            hi = 10.0
        return float(_brentq(_obj, lo, hi, xtol=tol, maxiter=max_iter))
    except Exception:
        return float("nan")


# ---- full BSM-Merton greeks (analytic) --------------------------------------
def bsm_greeks(S: float, K: float, T: float, r: float, q: float, sigma: float,
               is_call: bool) -> Dict[str, float]:
    """Full Black-Scholes-Merton greeks with continuous dividend yield q.

    Returns first-order (delta, vega, theta, rho), second-order (gamma, vanna,
    volga) and third-order (charm, speed, color) sensitivities. All analytic.
    Reference: Merton (1973); standard greek identities (Haug, Wilmott).
    """
    out = {k: float("nan") for k in
           ["delta", "gamma", "vega", "theta", "rho", "vanna", "volga",
            "charm", "speed", "color"]}
    if not (T > 0 and S > 0 and K > 0 and sigma > 0):
        return out
    sqrtT = math.sqrt(T)
    d1 = (math.log(S / K) + (r - q + 0.5 * sigma ** 2) * T) / (sigma * sqrtT)
    d2 = d1 - sigma * sqrtT
    Nd1 = _norm.cdf(d1); Nd2 = _norm.cdf(d2)
    nd1 = _norm_pdf(d1)
    eqt = math.exp(-q * T); ert = math.exp(-r * T)
    sign = 1.0 if is_call else -1.0
    # first order
    out["delta"] = sign * eqt * _norm.cdf(sign * d1)
    out["vega"] = S * eqt * nd1 * sqrtT
    out["rho"] = sign * K * T * ert * _norm.cdf(sign * d2)
    theta = (-(S * eqt * nd1 * sigma) / (2.0 * sqrtT)
             - sign * r * K * ert * _norm.cdf(sign * d2)
             + sign * q * S * eqt * _norm.cdf(sign * d1))
    out["theta"] = theta
    # second order
    out["gamma"] = eqt * nd1 / (S * sigma * sqrtT)
    out["vanna"] = -eqt * nd1 * d2 / sigma            # dVega/dSpot = dDelta/dVol
    out["volga"] = S * eqt * nd1 * sqrtT * d1 * d2 / sigma   # dVega/dVol
    # third order
    out["charm"] = (sign * q * eqt * _norm.cdf(sign * d1)
                    - eqt * nd1 * (2.0 * (r - q) * T - d2 * sigma * sqrtT)
                    / (2.0 * T * sigma * sqrtT))         # dDelta/dTime
    out["speed"] = -out["gamma"] / S * (d1 / (sigma * sqrtT) + 1.0)  # dGamma/dSpot
    # color = dGamma/d(calendar time) = -dGamma/dT (matches charm's time sign)
    out["color"] = (eqt * nd1 / (2.0 * S * T * sigma * sqrtT) *
                    (2.0 * q * T + 1.0 +
                     (2.0 * (r - q) * T - d2 * sigma * sqrtT)
                     / (sigma * sqrtT) * d1))            # dGamma/d(calendar t)
    # convention: vega/rho per 1.00 vol/rate; theta/charm/color per year
    return out


def enrich_with_iv_and_greeks(options: pd.DataFrame,
                              config: Dict[str, Any]) -> pd.DataFrame:
    """Populate canonical IV / greeks / forward / discount columns for every
    option row using the Let's-Be-Rational solver and analytic BSM greeks.

    Forwards are built from per-row r, q (parity-implied q where both a call and
    a put exist at the same strike/expiry; flat fallback otherwise)."""
    o = options.copy()
    if not len(o):
        return o
    rd = config["rates_dividends_forwards"]
    r_default = rd["fallback_flat_rate"]
    q_default = rd["fallback_dividend_yield"]
    o["risk_free_rate"] = o["risk_free_rate"].fillna(r_default).astype(float)
    o["dividend_yield"] = o["dividend_yield"].fillna(q_default).astype(float)
    o["T"] = o["days_to_expiry"].astype(float) / 365.0
    S = o["underlying_price"].astype(float).values
    Kv = o["strike"].astype(float).values
    rv = o["risk_free_rate"].values
    qv = o["dividend_yield"].values
    Tv = np.clip(o["T"].values, 1e-6, None)
    price = o["mid"].astype(float).values
    is_call = (o["option_type"].astype(str).values == "C")
    fwd = S * np.exp((rv - qv) * Tv)
    df_disc = np.exp(-rv * Tv)
    o["forward"] = fwd
    o["discount_factor"] = df_disc
    ivs = np.empty(len(o)); ivs[:] = np.nan
    greek_cols = ["delta", "gamma", "vega", "theta", "rho", "vanna",
                  "volga", "charm", "speed", "color"]
    gmat = {g: np.full(len(o), np.nan) for g in greek_cols}
    for i in range(len(o)):
        iv = implied_vol_lets_be_rational(price[i], fwd[i], Kv[i], Tv[i],
                                          rv[i], bool(is_call[i]),
                                          tol=config["iv_and_greeks"]["iv_tolerance"],
                                          max_iter=config["iv_and_greeks"]["iv_max_iterations"])
        ivs[i] = iv
        if np.isfinite(iv) and iv > 0:
            g = bsm_greeks(S[i], Kv[i], Tv[i], rv[i], qv[i], iv, bool(is_call[i]))
            for k in greek_cols:
                gmat[k][i] = g[k]
    o["implied_volatility"] = ivs
    for k in greek_cols:
        o[k] = gmat[k]
    # delta bucket on |delta|
    buckets = config["feature_engineering"]["delta_buckets"]
    o["delta_bucket"] = o["delta"].abs().apply(
        lambda d: assign_delta_bucket(d, buckets))
    return o


# ---- SSVI surface fit with no-arbitrage diagnostics -------------------------
# Reference: Gatheral & Jacquier, "Arbitrage-free SVI volatility surfaces"
# (2014), https://arxiv.org/abs/1204.0646 . We fit the surface-SVI (SSVI)
# parameterization in total-variance / log-moneyness space:
#   w(k, theta) = (theta/2) * (1 + rho*phi*k + sqrt((phi*k + rho)^2 + (1-rho^2)))
# with curvature phi(theta) = eta / (theta^gamma * (1+theta)^(1-gamma))
# (the "power-law" heston-like form). theta_t = ATM total variance per slice.
# No-arbitrage: butterfly (Gatheral-Jacquier Thm 4.2 sufficient conditions:
#   theta*phi*(1+|rho|) < 4  and  theta*phi^2*(1+|rho|) <= 4) and calendar
# monotonicity (theta non-decreasing in T => w non-decreasing in T at each k).

@dataclass
class SSVIParams:
    rho: float
    eta: float
    gamma: float
    thetas: Dict[float, float] = field(default_factory=dict)  # T -> ATM total var
    fit_rmse: float = float("nan")
    n_quotes: int = 0


def _ssvi_phi(theta: np.ndarray, eta: float, gamma: float) -> np.ndarray:
    theta = np.clip(theta, 1e-8, None)
    return eta / (np.power(theta, gamma) * np.power(1.0 + theta, 1.0 - gamma))


def ssvi_total_variance(k: np.ndarray, theta: float, rho: float,
                        eta: float, gamma: float) -> np.ndarray:
    phi = float(_ssvi_phi(np.array([theta]), eta, gamma)[0])
    a = phi * k + rho
    return (theta / 2.0) * (1.0 + rho * phi * k + np.sqrt(a * a + (1.0 - rho * rho)))


def fit_ssvi_surface(slice_df: pd.DataFrame,
                     config: Dict[str, Any]) -> Tuple[SSVIParams, pd.DataFrame]:
    """Fit one SSVI surface to a single trade_date's option slice.

    Expects columns: T (years), log_moneyness, implied_volatility, vega, bid, ask.
    Uses vega*(bid-ask)^-1 weighting per CONFIG.surface_fitting.quote_weighting.
    Returns (SSVIParams, per-quote diagnostics frame with model_iv, residual).
    """
    sf = config["surface_fitting"]
    d = slice_df.dropna(subset=["implied_volatility", "T", "log_moneyness"]).copy()
    d = d[(d["implied_volatility"] > 0) & (d["T"] > 0)]
    if len(d) < 6:
        return SSVIParams(rho=-0.3, eta=1.0, gamma=0.5, n_quotes=len(d)), d
    # winsorize IV
    lo, hi = sf["winsorize_iv_quantiles"]
    qv = d["implied_volatility"].quantile([lo, hi]).values
    d = d[d["implied_volatility"].between(qv[0], qv[1])]
    d["w_obs"] = (d["implied_volatility"] ** 2) * d["T"]      # observed total var
    # per-slice ATM theta (interpolate w at k=0 per expiry)
    thetas: Dict[float, float] = {}
    for T, g in d.groupby("T"):
        gg = g.sort_values("log_moneyness")
        try:
            theta = float(np.interp(0.0, gg["log_moneyness"], gg["w_obs"]))
        except Exception:
            theta = float(gg["w_obs"].median())
        thetas[float(T)] = max(theta, 1e-6)
    d["theta"] = d["T"].map(thetas)
    # quote weights
    if sf["quote_weighting"] == "vega_bidask":
        width = (d["ask"].astype(float) - d["bid"].astype(float)).clip(lower=1e-4)
        d["weight"] = (d["vega"].abs().fillna(1.0) / width).clip(1e-6, 1e6)
    else:
        d["weight"] = 1.0
    k = d["log_moneyness"].values
    th = d["theta"].values
    w_obs = d["w_obs"].values
    wgt = d["weight"].values

    def _resid(params):
        rho, eta, gamma = params
        phi = _ssvi_phi(th, eta, gamma)
        a = phi * k + rho
        w_model = (th / 2.0) * (1.0 + rho * phi * k + np.sqrt(a * a + (1.0 - rho * rho)))
        return np.sqrt(wgt) * (w_model - w_obs)

    from scipy.optimize import least_squares
    best = None
    for x0 in [(-0.3, 1.0, 0.5), (-0.6, 0.5, 0.4), (-0.1, 2.0, 0.3)]:
        try:
            res = least_squares(_resid, x0, bounds=([-0.999, 1e-3, 0.0],
                                                    [0.999, 50.0, 1.0]),
                                max_nfev=2000)
            if best is None or res.cost < best.cost:
                best = res
        except Exception:
            continue
    if best is None:
        return SSVIParams(rho=-0.3, eta=1.0, gamma=0.5, thetas=thetas,
                          n_quotes=len(d)), d
    rho, eta, gamma = best.x
    phi = _ssvi_phi(th, eta, gamma)
    a = phi * k + rho
    w_model = (th / 2.0) * (1.0 + rho * phi * k + np.sqrt(a * a + (1.0 - rho * rho)))
    d["model_total_var"] = w_model
    d["model_iv"] = np.sqrt(np.clip(w_model / d["T"].values, 0, None))
    d["iv_residual"] = d["model_iv"] - d["implied_volatility"]
    rmse = float(np.sqrt(np.mean(d["iv_residual"] ** 2)))
    params = SSVIParams(rho=float(rho), eta=float(eta), gamma=float(gamma),
                        thetas=thetas, fit_rmse=rmse, n_quotes=len(d))
    return params, d


def ssvi_no_arbitrage_diagnostics(params: SSVIParams) -> Dict[str, Any]:
    """Butterfly + calendar no-arbitrage diagnostics for a fitted SSVI surface.

    Butterfly (Gatheral-Jacquier sufficient conditions per slice theta):
        C1: theta * phi * (1 + |rho|) < 4
        C2: theta * phi^2 * (1 + |rho|) <= 4
    Calendar: ATM total variance theta(T) must be non-decreasing in T.
    """
    diag: Dict[str, Any] = {"butterfly_ok": True, "calendar_ok": True,
                            "violations": []}
    if not params.thetas:
        diag["butterfly_ok"] = diag["calendar_ok"] = False
        diag["violations"].append("no_thetas")
        return diag
    abs_rho = abs(params.rho)
    for T, theta in params.thetas.items():
        phi = float(_ssvi_phi(np.array([theta]), params.eta, params.gamma)[0])
        c1 = theta * phi * (1.0 + abs_rho)
        c2 = theta * phi * phi * (1.0 + abs_rho)
        if not (c1 < 4.0):
            diag["butterfly_ok"] = False
            diag["violations"].append(f"butterfly_C1@T={T:.3f}:{c1:.3f}>=4")
        if not (c2 <= 4.0):
            diag["butterfly_ok"] = False
            diag["violations"].append(f"butterfly_C2@T={T:.3f}:{c2:.3f}>4")
    Ts = sorted(params.thetas)
    th_series = [params.thetas[t] for t in Ts]
    for i in range(1, len(th_series)):
        if th_series[i] + 1e-9 < th_series[i - 1]:
            diag["calendar_ok"] = False
            diag["violations"].append(
                f"calendar@T={Ts[i]:.3f}:theta_decreasing")
    diag["max_butterfly_C1"] = max(
        (params.thetas[t] * float(_ssvi_phi(np.array([params.thetas[t]]),
                                            params.eta, params.gamma)[0])
         * (1.0 + abs_rho)) for t in Ts)
    return diag

def clean_and_align(store: "UnifiedMarketDataStore") -> Dict[str, pd.DataFrame]:
    """Produce a clean, aligned set of panels keyed by trade_date.

    - underlying: business-day OHLC for the primary underlying
    - aux: VIX family + put/call + rates/divs
    - options: validated canonical option panel (already gate-filtered)
    Returns a dict; also computes a per-day data-quality scalar in [0,1].
    """
    u = store.get_underlying_panel().copy()
    aux = store.get_auxiliary_panel().copy()
    o = store.get_option_panel().copy()
    for df in (u, aux, o):
        if len(df):
            df["trade_date"] = pd.to_datetime(df["trade_date"]).dt.normalize()
    if len(u):
        u = u.sort_values("trade_date").drop_duplicates("trade_date")
        u = u.set_index("trade_date")
    if len(aux):
        aux = aux.sort_values("trade_date").drop_duplicates("trade_date").set_index("trade_date")
    # per-day option count -> data-quality proxy
    if len(o):
        cnt = o.groupby("trade_date").size().rename("n_opts")
        dq = (cnt / cnt.quantile(0.9)).clip(0, 1).rename("data_quality_score")
    else:
        dq = pd.Series(dtype=float, name="data_quality_score")
    return {"underlying": u, "aux": aux, "options": o, "data_quality": dq}

# Realized-vol estimators (Section A.8). Yang-Zhang is the drift/jump-robust
# primary; close-to-close, Parkinson, Garman-Klass provided for completeness.
def rv_close_to_close(close: pd.Series, window: int) -> pd.Series:
    r = np.log(close / close.shift(1))
    return r.rolling(window).std() * math.sqrt(252.0)


def rv_parkinson(high: pd.Series, low: pd.Series, window: int) -> pd.Series:
    hl = (np.log(high / low)) ** 2
    factor = 1.0 / (4.0 * math.log(2.0))
    return np.sqrt(factor * hl.rolling(window).mean() * 252.0)


def rv_garman_klass(o, h, l, c, window: int) -> pd.Series:
    term = 0.5 * (np.log(h / l)) ** 2 - (2 * math.log(2) - 1) * (np.log(c / o)) ** 2
    return np.sqrt(term.rolling(window).mean() * 252.0)


def rv_yang_zhang(o, h, l, c, window: int) -> pd.Series:
    """Yang-Zhang (2000) drift-independent, jump-robust realized vol.
    Reference: https://www.jstor.org/stable/10.1086/209650"""
    log_oc = np.log(o / c.shift(1))      # overnight
    log_co = np.log(c / o)               # open->close
    log_ho = np.log(h / o); log_lo = np.log(l / o); log_cc = np.log(c / o)
    n = window
    sigma_o2 = log_oc.rolling(n).var()
    sigma_c2 = log_co.rolling(n).var()
    rs = log_ho * (log_ho - log_cc) + log_lo * (log_lo - log_cc)   # Rogers-Satchell
    sigma_rs2 = rs.rolling(n).mean()
    k = 0.34 / (1.34 + (n + 1) / (n - 1))
    yz = sigma_o2 + k * sigma_c2 + (1 - k) * sigma_rs2
    return np.sqrt(yz.clip(lower=0) * 252.0)


def compute_realized_vol_panel(u: pd.DataFrame,
                               config: Dict[str, Any]) -> pd.DataFrame:
    windows = [int(w.rstrip("D")) for w in
               config["feature_engineering"]["realized_vol_windows"]]
    out = pd.DataFrame(index=u.index)
    o, h, l, c = u["open"], u["high"], u["low"], u["close"]
    for w in windows:
        out[f"rv_cc_{w}"] = rv_close_to_close(c, w)
        out[f"rv_park_{w}"] = rv_parkinson(h, l, w)
        out[f"rv_gk_{w}"] = rv_garman_klass(o, h, l, c, w)
        out[f"rv_yz_{w}"] = rv_yang_zhang(o, h, l, c, w)
    return out


# ---- option-surface-derived daily features ----------------------------------
def _atm_iv_for_tenor(slice_df: pd.DataFrame, target_days: float) -> float:
    """Vega-weighted ATM IV near a target tenor (interp across |moneyness-1|)."""
    d = slice_df.dropna(subset=["implied_volatility"]).copy()
    if not len(d):
        return np.nan
    d["tdiff"] = (d["days_to_expiry"].astype(float) - target_days).abs()
    nearest = d.sort_values("tdiff").head(200)
    exp = nearest["days_to_expiry"].mode()
    if len(exp):
        nearest = nearest[nearest["days_to_expiry"] == exp.iloc[0]]
    nearest = nearest.assign(mdist=(nearest["moneyness"] - 1.0).abs())
    atm = nearest.sort_values("mdist").head(6)
    if not len(atm):
        return np.nan
    w = atm["vega"].abs().fillna(1.0) + 1e-9
    return float(np.average(atm["implied_volatility"], weights=w))


def _delta_iv(slice_df: pd.DataFrame, target_delta: float,
              option_type: str, target_days: float) -> float:
    """IV at a target |delta| for calls or puts near a tenor."""
    d = slice_df[(slice_df["option_type"] == option_type)].dropna(
        subset=["implied_volatility", "delta"]).copy()
    if not len(d):
        return np.nan
    d["tdiff"] = (d["days_to_expiry"].astype(float) - target_days).abs()
    exp = d.sort_values("tdiff")["days_to_expiry"]
    if len(exp):
        d = d[d["days_to_expiry"] == exp.iloc[0]]
    d["ddist"] = (d["delta"].abs() - target_delta).abs()
    row = d.sort_values("ddist").head(1)
    return float(row["implied_volatility"].iloc[0]) if len(row) else np.nan


def bkm_risk_neutral_moments(slice_df: pd.DataFrame,
                             target_days: float = 30.0) -> Dict[str, float]:
    """Bakshi-Kapadia-Madan (2003) model-free risk-neutral variance, skewness,
    kurtosis from the OTM option cross-section at the nearest tenor.
    Reference: http://www.ssrn.com/abstract=2731103 (BKM)."""
    d = slice_df.dropna(subset=["mid", "strike", "implied_volatility"]).copy()
    if not len(d):
        return {"rn_var": np.nan, "rn_skew": np.nan, "rn_kurt": np.nan}
    d["tdiff"] = (d["days_to_expiry"].astype(float) - target_days).abs()
    exp = d.sort_values("tdiff")["days_to_expiry"].iloc[0]
    d = d[d["days_to_expiry"] == exp]
    if d.empty:
        return {"rn_var": np.nan, "rn_skew": np.nan, "rn_kurt": np.nan}
    F = float(d["forward"].dropna().median()) if d["forward"].notna().any() \
        else float(d["underlying_price"].iloc[0])
    T = float(exp) / 365.0
    r = float(d["risk_free_rate"].iloc[0])
    # OTM calls (K>=F) and OTM puts (K<F)
    calls = d[(d["option_type"] == "C") & (d["strike"] >= F)].sort_values("strike")
    puts = d[(d["option_type"] == "P") & (d["strike"] < F)].sort_values("strike")
    if len(calls) < 2 and len(puts) < 2:
        return {"rn_var": np.nan, "rn_skew": np.nan, "rn_kurt": np.nan}
    Ks = np.concatenate([puts["strike"].values, calls["strike"].values])
    Qs = np.concatenate([puts["mid"].values, calls["mid"].values])
    order = np.argsort(Ks); Ks = Ks[order]; Qs = Qs[order]
    if len(Ks) < 3:
        return {"rn_var": np.nan, "rn_skew": np.nan, "rn_kurt": np.nan}
    dK = np.gradient(Ks)
    # BKM contract weights (log-contract derivatives)
    x = np.log(Ks / F)
    w_v = (2.0 * (1.0 - x)) / (Ks ** 2)          # variance contract
    w_s = (6.0 * x - 3.0 * x ** 2) / (Ks ** 2)    # cubic contract
    w_k = (12.0 * x ** 2 - 4.0 * x ** 3) / (Ks ** 2)
    erT = math.exp(r * T)
    V = float(np.sum(w_v * Qs * dK) * erT)
    W = float(np.sum(w_s * Qs * dK) * erT)
    X = float(np.sum(w_k * Qs * dK) * erT)
    mu = erT - 1.0 - erT / 2.0 * V - erT / 6.0 * W - erT / 24.0 * X
    var = V - mu ** 2
    if var <= 0:
        return {"rn_var": float(max(var, 0)), "rn_skew": np.nan, "rn_kurt": np.nan}
    skew = (erT * W - 3.0 * mu * erT * V + 2.0 * mu ** 3) / (var ** 1.5)
    kurt = (erT * X - 4.0 * mu * erT * W + 6.0 * mu ** 2 * erT * V
            - 3.0 * mu ** 4) / (var ** 2)
    return {"rn_var": float(var), "rn_skew": float(skew), "rn_kurt": float(kurt)}


def dealer_exposures(slice_df: pd.DataFrame,
                     target_days: float = 30.0) -> Dict[str, float]:
    """Dealer gamma/vanna/charm exposure proxies with explicit sign convention.

    Convention (SpotGamma-style): dealers are assumed long calls / short puts
    from customer flow => signed gamma = +call_gamma*OI - put_gamma*OI scaled by
    spot^2 * multiplier. gamma_flip is the spot where net GEX crosses zero
    (here approximated by the OI-weighted gamma-neutral strike). call_wall /
    put_wall = strikes of max call / put gamma*OI.
    Reference: https://spotgamma.com/vanna-and-charm-explained/"""
    d = slice_df.dropna(subset=["gamma", "open_interest"]).copy()
    if not len(d):
        return {k: np.nan for k in ["GEX", "vanna_exposure", "charm_exposure",
                                    "gamma_flip_level", "call_wall", "put_wall"]}
    d["tdiff"] = (d["days_to_expiry"].astype(float) - target_days).abs()
    exp = d.sort_values("tdiff")["days_to_expiry"].iloc[0]
    d = d[d["days_to_expiry"] == exp]
    S = float(d["underlying_price"].iloc[0])
    mult = float(d["contract_multiplier"].iloc[0]) if d["contract_multiplier"].notna().any() else 100.0
    sign = np.where(d["option_type"].values == "C", 1.0, -1.0)
    oi = d["open_interest"].fillna(0).astype(float).values
    gex_contrib = sign * d["gamma"].values * oi * (S ** 2) * mult * 0.01
    vanna_contrib = sign * d["vanna"].fillna(0).values * oi * S * mult
    charm_contrib = sign * d["charm"].fillna(0).values * oi * S * mult
    # walls by gamma*OI per strike
    d2 = d.assign(gOI=(d["gamma"] * oi).values)
    call_wall = d2[d2["option_type"] == "C"].groupby("strike")["gOI"].sum()
    put_wall = d2[d2["option_type"] == "P"].groupby("strike")["gOI"].sum()
    # gamma flip: spot-weighted strike where net signed gamma*OI changes sign
    by_strike = d.assign(sgOI=(sign * d["gamma"].values * oi)).groupby("strike")["sgOI"].sum().sort_index()
    cum = by_strike.cumsum()
    flip = np.nan
    if len(cum) > 1:
        sgn = np.sign(cum.values)
        idx = np.where(np.diff(sgn) != 0)[0]
        if len(idx):
            flip = float(cum.index[idx[0]])
    return {
        "GEX": float(np.nansum(gex_contrib)),
        "vanna_exposure": float(np.nansum(vanna_contrib)),
        "charm_exposure": float(np.nansum(charm_contrib)),
        "gamma_flip_level": flip,
        "call_wall": float(call_wall.idxmax()) if len(call_wall) else np.nan,
        "put_wall": float(put_wall.idxmax()) if len(put_wall) else np.nan,
    }


def robust_rolling_zscore(s: pd.Series, window: int = 252,
                          min_periods: int = 60) -> pd.Series:
    """Leakage-safe robust rolling z-score using trailing median / MAD ONLY.
    Uses data up to and including t (no future data). Section B
    standardization_method=robust_rolling_zscore."""
    med = s.rolling(window, min_periods=min_periods).median()
    mad = (s - med).abs().rolling(window, min_periods=min_periods).median()
    scale = 1.4826 * mad
    return (s - med) / scale.replace(0, np.nan)


def expanding_inseample_zscore(s: pd.Series, min_periods: int = 60) -> pd.Series:
    """Expanding-window z-score (mean/std up to t only). Leakage-safe
    replacement for v4c's full-sample zscore() in time-series use."""
    mu = s.expanding(min_periods=min_periods).mean()
    sd = s.expanding(min_periods=min_periods).std()
    return (s - mu) / sd.replace(0, np.nan)


def build_feature_matrix(panels: Dict[str, pd.DataFrame],
                         enriched_options: pd.DataFrame,
                         config: Dict[str, Any]) -> pd.DataFrame:
    """Assemble the full daily feature matrix (one row per trade_date).

    Combines realized-vol estimators, ATM/skew/term IV features, canonical VRP
    (IV^2 - RV^2), BKM risk-neutral moments, VIX-complex features, dealer
    exposures, and put/call ratios. All time-series standardization uses
    leakage-safe trailing windows. Returns features indexed by trade_date.
    """
    fe = config["feature_engineering"]
    u = panels["underlying"]; aux = panels["aux"]
    rv = compute_realized_vol_panel(u, config)
    feats = pd.DataFrame(index=u.index)
    feats = feats.join(rv)

    # per-date option-surface features
    opt = enriched_options.copy()
    opt["trade_date"] = pd.to_datetime(opt["trade_date"]).dt.normalize()
    surf_rows = []
    for tdate, sl in opt.groupby("trade_date"):
        atm_1m = _atm_iv_for_tenor(sl, 30)
        atm_2m = _atm_iv_for_tenor(sl, 60)
        atm_3m = _atm_iv_for_tenor(sl, 90)
        put25 = _delta_iv(sl, 0.25, "P", 30)
        call25 = _delta_iv(sl, 0.25, "C", 30)
        put10 = _delta_iv(sl, 0.10, "P", 30)
        call10 = _delta_iv(sl, 0.10, "C", 30)
        bkm = bkm_risk_neutral_moments(sl, 30)
        de = dealer_exposures(sl, 30)
        # SSVI params per day (curvature features)
        params, _ = fit_ssvi_surface(sl.assign(T=sl["days_to_expiry"].astype(float)/365.0), config)
        diag = ssvi_no_arbitrage_diagnostics(params)
        surf_rows.append({
            "trade_date": tdate, "atm_iv_1m": atm_1m, "atm_iv_2m": atm_2m,
            "atm_iv_3m": atm_3m,
            "skew_put25_call25": (put25 - call25) if np.isfinite(put25) and np.isfinite(call25) else np.nan,
            "skew_put10_call10": (put10 - call10) if np.isfinite(put10) and np.isfinite(call10) else np.nan,
            "iv_3m_minus_1m": (atm_3m - atm_1m) if np.isfinite(atm_3m) and np.isfinite(atm_1m) else np.nan,
            "rn_var": bkm["rn_var"], "rn_skew": bkm["rn_skew"], "rn_kurt": bkm["rn_kurt"],
            "GEX": de["GEX"], "vanna_exposure": de["vanna_exposure"],
            "charm_exposure": de["charm_exposure"],
            "gamma_flip_level": de["gamma_flip_level"],
            "call_wall": de["call_wall"], "put_wall": de["put_wall"],
            "ssvi_rho": params.rho, "ssvi_eta": params.eta,
            "ssvi_gamma": params.gamma, "ssvi_rmse": params.fit_rmse,
            "ssvi_no_arb_ok": bool(diag["butterfly_ok"] and diag["calendar_ok"]),
        })
    surf = pd.DataFrame(surf_rows).set_index("trade_date") if surf_rows else pd.DataFrame()
    feats = feats.join(surf)

    # VIX complex + put/call
    if len(aux):
        for col in ["VIX", "VVIX", "VIX9D", "VIX3M", "VIX6M", "VIX1Y",
                    "total_put_call", "index_put_call", "vix_put_call"]:
            if col in aux.columns:
                feats[col] = aux[col]
        if "VIX" in aux.columns and "VIX3M" in aux.columns:
            feats["vix_term_slope"] = aux["VIX3M"] - aux["VIX"]

    # canonical VRP: IV^2 - RV^2 (annualized variance points). AQR/Carr-Wu sign.
    if "atm_iv_1m" in feats and "rv_yz_21" in feats:
        feats["vrp_1m"] = feats["atm_iv_1m"] ** 2 - feats["rv_yz_21"] ** 2
    if "VIX" in feats and "rv_yz_21" in feats:
        feats["vrp_vix"] = (feats["VIX"] / 100.0) ** 2 - feats["rv_yz_21"] ** 2

    # term-structure: IV_1m - RV_21d
    if "atm_iv_1m" in feats and "rv_yz_21" in feats:
        feats["iv1m_minus_rv21"] = feats["atm_iv_1m"] - feats["rv_yz_21"]

    feats = feats.sort_index()
    return feats

# Gaussian HMM implemented in-module (Baum-Welch EM + scaled forward-backward).
# We expose BOTH filtered (alpha-only, causal) and smoothed (alpha-beta) state
# posteriors. Section B hard constraint: FILTERED probabilities drive all live
# decisions; smoothed are diagnostics only. Reference: Hamilton (regime
# switching, filtered vs smoothed) https://econweb.ucsd.edu/~jhamilto/palgrav1.pdf

class GaussianHMM:
    def __init__(self, n_states: int, n_iter: int = 100, tol: float = 1e-4,
                 cov_type: str = "full", seed: int = 1832,
                 reg_covar: float = 1e-4):
        self.K = n_states
        self.n_iter = n_iter
        self.tol = tol
        self.cov_type = cov_type
        self.reg = reg_covar
        self.rng = np.random.default_rng(seed)
        self.pi = None; self.A = None; self.mu = None; self.cov = None
        self.loglik_ = -np.inf
        self.fitted_ = False

    def _mvn_logpdf(self, X):
        from scipy.stats import multivariate_normal as mvn
        n, d = X.shape
        logB = np.empty((n, self.K))
        for k in range(self.K):
            cov = self.cov[k]
            if self.cov_type == "diag":
                cov = np.diag(np.diag(cov))
            logB[:, k] = mvn.logpdf(X, mean=self.mu[k],
                                    cov=cov + self.reg * np.eye(d),
                                    allow_singular=True)
        return logB

    def _init(self, X):
        from sklearn.cluster import KMeans
        n, d = X.shape
        km = KMeans(self.K, n_init=5, random_state=1832).fit(X)
        lab = km.labels_
        self.mu = km.cluster_centers_.copy()
        self.cov = np.array([np.cov(X[lab == k].T) + self.reg * np.eye(d)
                             if (lab == k).sum() > d else np.eye(d)
                             for k in range(self.K)])
        self.pi = np.full(self.K, 1.0 / self.K)
        self.A = np.full((self.K, self.K), 1.0 / self.K)

    def fit(self, X):
        X = np.asarray(X, float)
        n, d = X.shape
        self._init(X)
        prev_ll = -np.inf
        for _ in range(self.n_iter):
            logB = self._mvn_logpdf(X)
            B = np.exp(logB - logB.max(axis=1, keepdims=True))
            # scaled forward
            alpha = np.zeros((n, self.K)); c = np.zeros(n)
            alpha[0] = self.pi * B[0]; c[0] = alpha[0].sum() + 1e-300
            alpha[0] /= c[0]
            for t in range(1, n):
                alpha[t] = (alpha[t - 1] @ self.A) * B[t]
                c[t] = alpha[t].sum() + 1e-300
                alpha[t] /= c[t]
            # scaled backward
            beta = np.zeros((n, self.K)); beta[-1] = 1.0
            for t in range(n - 2, -1, -1):
                beta[t] = (self.A @ (B[t + 1] * beta[t + 1])) / c[t + 1]
            gamma = alpha * beta
            gamma /= gamma.sum(axis=1, keepdims=True) + 1e-300
            # xi
            xi = np.zeros((self.K, self.K))
            for t in range(n - 1):
                num = (alpha[t][:, None] * self.A *
                       (B[t + 1] * beta[t + 1])[None, :]) / c[t + 1]
                xi += num / (num.sum() + 1e-300)
            # M-step
            self.pi = gamma[0] / gamma[0].sum()
            self.A = xi / (xi.sum(axis=1, keepdims=True) + 1e-300)
            Nk = gamma.sum(axis=0) + 1e-300
            self.mu = (gamma.T @ X) / Nk[:, None]
            for k in range(self.K):
                dx = X - self.mu[k]
                self.cov[k] = (gamma[:, k][:, None] * dx).T @ dx / Nk[k] \
                    + self.reg * np.eye(d)
            ll = np.sum(np.log(c))
            if abs(ll - prev_ll) < self.tol:
                prev_ll = ll
                break
            prev_ll = ll
        self.loglik_ = prev_ll
        self.fitted_ = True
        self._n, self._d = n, d
        return self

    def n_params(self):
        d = self._d
        cov_params = self.K * (d * (d + 1) // 2 if self.cov_type == "full" else d)
        return (self.K - 1) + self.K * (self.K - 1) + self.K * d + cov_params

    def bic(self):
        return -2.0 * self.loglik_ + self.n_params() * math.log(self._n)

    def filtered(self, X):
        """Causal filtered posteriors P(state_t | x_1..x_t). LIVE decisions."""
        X = np.asarray(X, float); n = len(X)
        logB = self._mvn_logpdf(X)
        B = np.exp(logB - logB.max(axis=1, keepdims=True))
        alpha = np.zeros((n, self.K)); c = np.zeros(n)
        alpha[0] = self.pi * B[0]; c[0] = alpha[0].sum() + 1e-300; alpha[0] /= c[0]
        for t in range(1, n):
            alpha[t] = (alpha[t - 1] @ self.A) * B[t]
            c[t] = alpha[t].sum() + 1e-300; alpha[t] /= c[t]
        return alpha

    def smoothed(self, X):
        """Non-causal smoothed posteriors P(state_t | x_1..x_T). DIAGNOSTIC."""
        X = np.asarray(X, float); n = len(X)
        logB = self._mvn_logpdf(X)
        B = np.exp(logB - logB.max(axis=1, keepdims=True))
        alpha = self.filtered(X)
        c = np.array([1.0] * n)
        beta = np.zeros((n, self.K)); beta[-1] = 1.0
        # recompute c for proper scaling
        a = np.zeros((n, self.K)); cc = np.zeros(n)
        a[0] = self.pi * B[0]; cc[0] = a[0].sum() + 1e-300; a[0] /= cc[0]
        for t in range(1, n):
            a[t] = (a[t - 1] @ self.A) * B[t]; cc[t] = a[t].sum() + 1e-300; a[t] /= cc[t]
        for t in range(n - 2, -1, -1):
            beta[t] = (self.A @ (B[t + 1] * beta[t + 1])) / cc[t + 1]
        g = a * beta
        g /= g.sum(axis=1, keepdims=True) + 1e-300
        return g


def select_hmm(X: np.ndarray, candidates: Sequence[int],
               cov_type: str = "full", seed: int = 1832) -> GaussianHMM:
    """Fit HMMs across candidate state counts; pick min BIC (Section B
    hmm_selection_metric=bic). Enforces min state occupancy on filtered probs."""
    best = None
    for k in candidates:
        try:
            hmm = GaussianHMM(k, cov_type=cov_type, seed=seed).fit(X)
            if best is None or hmm.bic() < best.bic():
                best = hmm
        except Exception:
            continue
    return best


# ---- change-point detection (PELT, L2 cost) ---------------------------------
# Reference: change points in VIX/S&P
# https://orbit.dtu.dk/files/127117501/Detecting_Change_Points_in_VIX_and_S_P_500_ACCEPTED_VERSION.pdf
def pelt_l2(y: np.ndarray, penalty: float, min_size: int = 10) -> List[int]:
    """Pruned Exact Linear Time change-point detection with an L2 (mean-shift)
    cost. Returns sorted change-point indices (exclusive ends)."""
    y = np.asarray(y, float)
    n = len(y)
    if n < 2 * min_size:
        return []
    cs = np.concatenate([[0], np.cumsum(y)])
    cs2 = np.concatenate([[0], np.cumsum(y * y)])

    def cost(a, b):
        m = b - a
        s = cs[b] - cs[a]; s2 = cs2[b] - cs2[a]
        return s2 - (s * s) / m

    F = np.full(n + 1, np.inf); F[0] = -penalty
    cp = {0: []}
    R = [0]
    for t in range(min_size, n + 1):
        best_val = np.inf; best_s = None
        for s in R:
            if t - s < min_size:
                continue
            val = F[s] + cost(s, t) + penalty
            if val < best_val:
                best_val = val; best_s = s
        if best_s is None:
            continue
        F[t] = best_val
        cp[t] = cp[best_s] + [best_s]
        R = [s for s in R if F[s] + cost(s, t) <= F[t]] + [t]
    pts = [p for p in cp.get(n, []) if 0 < p < n]
    return sorted(set(pts))


# ---- clustering ensemble -----------------------------------------------------
def cluster_states(X: np.ndarray, methods: Sequence[str],
                   n_candidates: Sequence[int], seed: int = 1832) -> Dict[str, Any]:
    """Fit GMM / KMeans / hierarchical clusterings; pick best by silhouette."""
    from sklearn.cluster import KMeans, AgglomerativeClustering
    from sklearn.mixture import GaussianMixture
    from sklearn.metrics import silhouette_score
    out = {}
    for meth in methods:
        best = None
        for k in n_candidates:
            try:
                if meth == "kmeans":
                    lab = KMeans(k, n_init=5, random_state=seed).fit_predict(X)
                elif meth == "gaussian_mixture":
                    lab = GaussianMixture(k, random_state=seed,
                                          reg_covar=1e-4).fit_predict(X)
                elif meth == "hierarchical":
                    lab = AgglomerativeClustering(k).fit_predict(X)
                else:
                    continue
                if len(set(lab)) < 2:
                    continue
                sil = silhouette_score(X, lab)
                if best is None or sil > best[0]:
                    best = (sil, k, lab)
            except Exception:
                continue
        if best is not None:
            out[meth] = {"silhouette": best[0], "k": best[1], "labels": best[2]}
    return out


# ---- rule-based classifier (VIX/VRP/skew/term) ------------------------------
def rule_based_regime(feats: pd.DataFrame,
                      regime_names: Sequence[str]) -> pd.Series:
    """Transparent rule classifier mapping daily features to the 6 spec regimes.
    Uses leakage-safe trailing percentiles for thresholds."""
    f = feats.copy()
    vix = f.get("VIX", pd.Series(index=f.index, dtype=float))
    vrp = f.get("vrp_1m", pd.Series(index=f.index, dtype=float))
    skew = f.get("skew_put25_call25", pd.Series(index=f.index, dtype=float))
    term = f.get("vix_term_slope", pd.Series(index=f.index, dtype=float))
    rv = f.get("rv_yz_21", pd.Series(index=f.index, dtype=float))
    # trailing percentile ranks (causal): fraction of trailing window <= last
    def rrank(s, w=252):
        return s.rolling(w, min_periods=40).apply(
            lambda x: float((x <= x[-1]).mean()), raw=True)
    vix_hi = rrank(vix) > 0.8
    vix_lo = rrank(vix) < 0.3
    skew_hi = rrank(skew) > 0.8
    backwardation = term < 0
    high_vrp = rrank(vrp) > 0.6
    labels = pd.Series("mixed_uncertain", index=f.index)
    labels[vix_lo & (term > 0) & high_vrp] = "low_vol_contango_high_vrp"
    labels[vix_hi & backwardation] = "vol_spike_backwardation"
    labels[skew_hi & ~vix_hi] = "high_skew_stress"
    labels[(rrank(vix) < 0.5) & (rv < rv.rolling(63, min_periods=20).mean())] = "mean_reverting_normalization"
    labels[(rrank(rv) > 0.5) & ~vix_hi & (skew.abs() < skew.abs().rolling(63, min_periods=20).median())] = "trending_low_gamma"
    return labels


# ---- regime labeling: map latent HMM states -> named regimes ----------------
def label_hmm_states(filtered: np.ndarray, feats: pd.DataFrame,
                     regime_names: Sequence[str]) -> Dict[int, str]:
    """Assign human-readable regime names to latent HMM states by each state's
    mean feature profile (VIX level, VRP, skew, term slope, RV)."""
    states = filtered.argmax(axis=1)
    prof = {}
    cols = [c for c in ["VIX", "vrp_1m", "skew_put25_call25",
                        "vix_term_slope", "rv_yz_21"] if c in feats.columns]
    for k in range(filtered.shape[1]):
        mask = states == k
        if mask.sum() == 0:
            prof[k] = {c: np.nan for c in cols}
        else:
            prof[k] = {c: float(np.nanmean(feats[c].values[mask])) for c in cols}
    # rank states by VIX to assign archetypes
    order = sorted(prof, key=lambda k: (prof[k].get("VIX", 0) if np.isfinite(prof[k].get("VIX", np.nan)) else 0))
    mapping = {}
    archetypes = ["low_vol_contango_high_vrp", "mean_reverting_normalization",
                  "trending_low_gamma", "high_skew_stress",
                  "vol_spike_backwardation", "mixed_uncertain"]
    for i, k in enumerate(order):
        mapping[k] = archetypes[min(i, len(archetypes) - 1)]
    return mapping


# ---- BMA ensemble fusion -----------------------------------------------------
def bma_ensemble(filtered_hmm: np.ndarray, rule_labels: pd.Series,
                 cluster_labels: Optional[np.ndarray],
                 hmm_state_map: Dict[int, str],
                 regime_names: Sequence[str],
                 weights: Tuple[float, float, float] = (0.5, 0.3, 0.2)
                 ) -> pd.DataFrame:
    """Bayesian-model-average fusion of HMM (filtered), rule, and cluster votes
    into a posterior over the 6 named regimes (Section B
    ensemble_fusion_method=bayesian_model_average)."""
    idx = rule_labels.index
    rn = list(regime_names)
    K = len(rn)
    post = np.zeros((len(idx), K))
    wh, wr, wc = weights
    # HMM contribution
    hmm_states = filtered_hmm.argmax(axis=1)
    for t in range(len(idx)):
        # HMM: distribute filtered prob mass onto named regimes
        for s in range(filtered_hmm.shape[1]):
            name = hmm_state_map.get(s, "mixed_uncertain")
            if name in rn:
                post[t, rn.index(name)] += wh * filtered_hmm[t, s]
        # rule vote
        rl = rule_labels.iloc[t]
        if rl in rn:
            post[t, rn.index(rl)] += wr
        # cluster vote (map cluster id by majority not available here -> uniform skip if None)
    if cluster_labels is not None and len(cluster_labels) == len(idx):
        # map each cluster to its modal rule label
        cl2name = {}
        for cl in np.unique(cluster_labels):
            sub = rule_labels.values[cluster_labels == cl]
            if len(sub):
                cl2name[cl] = pd.Series(sub).mode().iloc[0]
        for t in range(len(idx)):
            name = cl2name.get(cluster_labels[t])
            if name in rn:
                post[t, rn.index(name)] += wc
    post = post / (post.sum(axis=1, keepdims=True) + 1e-12)
    df = pd.DataFrame(post, index=idx, columns=rn)
    df["regime"] = df[rn].idxmax(axis=1)
    return df


def build_regime_feature_frame(feats: pd.DataFrame,
                               config: Dict[str, Any]) -> pd.DataFrame:
    """Construct the causal, leakage-safe regime feature matrix restricted to
    option-snapshot dates (where surface features exist), robust-rolling
    z-scored. These features feed the HMM / clustering / rule engines."""
    keys = [k for k in ["VIX", "vrp_1m", "skew_put25_call25",
                        "vix_term_slope", "rv_yz_21", "rn_skew", "GEX"]
            if k in feats.columns]
    sub = feats[feats["atm_iv_1m"].notna()].copy() if "atm_iv_1m" in feats else feats.copy()
    Z = pd.DataFrame({k: robust_rolling_zscore(sub[k], window=104,
                                               min_periods=30)
                      for k in keys}, index=sub.index)
    return Z.dropna()


def run_regime_engine(feats: pd.DataFrame,
                      config: Dict[str, Any]) -> Dict[str, Any]:
    """Full Section-07 regime engine: HMM (filtered+smoothed, BIC-selected),
    change-points, clustering, rule classifier, BMA ensemble. Returns a dict of
    all artifacts. FILTERED HMM probabilities drive the ensemble (live)."""
    re_cfg = config["regime_engine"]
    Z = build_regime_feature_frame(feats, config)
    if len(Z) < 60:
        raise RuntimeError("Insufficient regime observations.")
    X = Z.values
    hmm = select_hmm(X, re_cfg["hmm_state_candidates"],
                     seed=config["project"]["random_seed"])
    filt = hmm.filtered(X)
    smo = hmm.smoothed(X)
    fsub = feats.loc[Z.index]
    smap = label_hmm_states(filt, fsub, re_cfg["regime_names"])
    rule = rule_based_regime(fsub, re_cfg["regime_names"])
    clusters = cluster_states(X, [mth for mth in re_cfg["clustering_methods"]
                                  if mth in ("kmeans", "gaussian_mixture",
                                             "hierarchical")],
                              re_cfg["clustering_n_candidates"],
                              seed=config["project"]["random_seed"])
    clab = None
    if "gaussian_mixture" in clusters:
        clab = clusters["gaussian_mixture"]["labels"]
    elif clusters:
        clab = list(clusters.values())[0]["labels"]
    ensemble = bma_ensemble(filt, rule, clab, smap, re_cfg["regime_names"])
    # change points on VIX (multiple penalties)
    cps = {}
    if "VIX" in fsub.columns:
        vv = fsub["VIX"].dropna().values
        for pen in re_cfg["changepoint_penalty_grid"]:
            cps[pen] = pelt_l2(vv, penalty=float(pen) * 25.0)
    return {
        "Z": Z, "hmm": hmm, "filtered": filt, "smoothed": smo,
        "state_map": smap, "rule_labels": rule, "clusters": clusters,
        "ensemble": ensemble, "change_points": cps,
        "regime_today": ensemble["regime"].iloc[-1] if len(ensemble) else None,
    }

# Full statistical-validation suite. All tests operate on per-trade or per-period
# return series. References:
#   Deflated Sharpe / PBO: Bailey & Lopez de Prado
#     http://www.ssrn.com/abstract=2568435 (PBO),
#     http://pm-research.com/lookup/doi/10.3905/jpm.2014.40.5.094 (DSR)
#   White Reality Check: http://doi.wiley.com/10.1111/1468-0262.00152
#   Hansen SPA: http://www.tandfonline.com/doi/abs/10.1198/073500105000000063

def sharpe_ratio(returns: np.ndarray, periods_per_year: float = 52.0) -> float:
    r = np.asarray(returns, float)
    r = r[np.isfinite(r)]
    if len(r) < 2 or r.std(ddof=1) == 0:
        return 0.0
    return float(r.mean() / r.std(ddof=1) * math.sqrt(periods_per_year))


def deflated_sharpe_ratio(returns: np.ndarray, n_trials: int,
                          periods_per_year: float = 52.0,
                          var_sr_trials: Optional[float] = None
                          ) -> Dict[str, float]:
    """Deflated Sharpe Ratio (Bailey & Lopez de Prado 2014). Adjusts the
    observed Sharpe for multiple testing, non-normality (skew/kurtosis) and
    sample length, returning a probabilistic (deflated) Sharpe in [0,1].

    var_sr_trials: variance of the (per-period) Sharpe ratios across the trials
    explored. When the empirical cross-strategy Sharpe variance is available it
    should be passed in; this yields a materially more honest (less optimistic)
    SR* hurdle than the 1/(n-1) sampling-variance fallback, which collapses to
    ~0 for long samples and inflates DSR for near-zero-edge strategies."""
    from scipy.stats import skew, kurtosis, norm
    r = np.asarray(returns, float); r = r[np.isfinite(r)]
    n = len(r)
    if n < 8 or r.std(ddof=1) == 0:
        return {"sr": 0.0, "dsr": 0.0, "sr_star": 0.0}
    sr = r.mean() / r.std(ddof=1)            # per-period (non-annualized)
    g3 = float(skew(r)); g4 = float(kurtosis(r, fisher=False))
    # Expected-maximum Sharpe under N independent trials (SR* benchmark).
    emc = 0.5772156649                       # Euler-Mascheroni
    N = max(int(n_trials), 1)
    if var_sr_trials is None or not np.isfinite(var_sr_trials) or var_sr_trials <= 0:
        var_sr_trials = 1.0 / max(n - 1, 1)
    if N > 1:
        z1 = norm.ppf(1.0 - 1.0 / N)
        z2 = norm.ppf(1.0 - 1.0 / N * math.e ** -1)
        sr_star = math.sqrt(var_sr_trials) * ((1 - emc) * z1 + emc * z2)
    else:
        sr_star = 0.0
    # Deflated Sharpe Ratio (probabilistic), non-normality-adjusted denominator.
    denom = math.sqrt(max(1.0 - g3 * sr + (g4 - 1.0) / 4.0 * sr ** 2, 1e-9))
    dsr = float(norm.cdf(((sr - sr_star) * math.sqrt(n - 1)) / denom))
    return {"sr": float(sr * math.sqrt(periods_per_year)),
            "dsr": dsr, "sr_star": float(sr_star)}


def probability_of_backtest_overfitting(returns_matrix: np.ndarray,
                                        n_splits: int = 10) -> float:
    """PBO via Combinatorially-Symmetric Cross-Validation (Bailey et al. 2015).
    returns_matrix: (T x N) of N strategy/config return series. Returns the
    probability that the in-sample-best config underperforms OOS (median rank)."""
    M = np.asarray(returns_matrix, float)
    T, N = M.shape
    if N < 2 or T < 2 * n_splits:
        return float("nan")
    idx = np.array_split(np.arange(T), n_splits)
    from itertools import combinations
    logits = []
    half = n_splits // 2
    for combo in combinations(range(n_splits), half):
        is_rows = np.concatenate([idx[i] for i in combo])
        oos_rows = np.concatenate([idx[i] for i in range(n_splits) if i not in combo])
        is_sr = M[is_rows].mean(0) / (M[is_rows].std(0, ddof=1) + 1e-12)
        oos_sr = M[oos_rows].mean(0) / (M[oos_rows].std(0, ddof=1) + 1e-12)
        n_star = int(np.argmax(is_sr))
        # rank of IS-best in OOS
        rank = (oos_sr <= oos_sr[n_star]).mean()
        rank = min(max(rank, 1e-6), 1 - 1e-6)
        logits.append(math.log(rank / (1 - rank)))
    logits = np.array(logits)
    return float((logits <= 0).mean())


def stationary_bootstrap_indices(n: int, avg_block: float,
                                 rng: np.random.Generator) -> np.ndarray:
    """Politis-Romano stationary bootstrap index sampler (geometric blocks)."""
    p = 1.0 / max(avg_block, 1.0)
    idx = np.empty(n, int)
    cur = rng.integers(0, n)
    for i in range(n):
        idx[i] = cur
        if rng.random() < p:
            cur = rng.integers(0, n)
        else:
            cur = (cur + 1) % n
    return idx


def white_reality_check(loss_diff: np.ndarray, n_boot: int = 1000,
                        avg_block: float = 5.0, seed: int = 1832) -> float:
    """White's Reality Check p-value. loss_diff: (T x N) per-period outperformance
    of N strategies over a benchmark (positive = better). Returns bootstrap
    p-value for H0: best strategy has no superior predictive ability."""
    D = np.asarray(loss_diff, float)
    if D.ndim == 1:
        D = D[:, None]
    T, N = D.shape
    rng = np.random.default_rng(seed)
    f_bar = D.mean(0)
    V = math.sqrt(T) * f_bar.max()
    boot_stats = np.empty(n_boot)
    for b in range(n_boot):
        ii = stationary_bootstrap_indices(T, avg_block, rng)
        fb = D[ii].mean(0)
        boot_stats[b] = math.sqrt(T) * (fb - f_bar).max()
    return float((boot_stats >= V).mean())


def hansen_spa(loss_diff: np.ndarray, n_boot: int = 1000,
               avg_block: float = 5.0, seed: int = 1832) -> float:
    """Hansen's SPA test p-value (studentized, consistent recentering).
    loss_diff: (T x N) outperformance over benchmark. Less conservative than
    White's RC by down-weighting poor models."""
    D = np.asarray(loss_diff, float)
    if D.ndim == 1:
        D = D[:, None]
    T, N = D.shape
    rng = np.random.default_rng(seed)
    f_bar = D.mean(0)
    omega = D.std(0, ddof=1) + 1e-12
    T_spa = max(np.max(math.sqrt(T) * f_bar / omega), 0.0)
    # consistent recentering threshold
    thresh = -omega * math.sqrt(2.0 * math.log(math.log(max(T, 3)))) / math.sqrt(T)
    g = np.where(f_bar >= thresh, f_bar, 0.0)
    boot_stats = np.empty(n_boot)
    for b in range(n_boot):
        ii = stationary_bootstrap_indices(T, avg_block, rng)
        fb = D[ii].mean(0)
        z = math.sqrt(T) * (fb - g) / omega
        boot_stats[b] = max(np.max(z), 0.0)
    return float((boot_stats >= T_spa).mean())


def benjamini_hochberg(pvals: Sequence[float], alpha: float = 0.10
                       ) -> Tuple[np.ndarray, float]:
    """Benjamini-Hochberg FDR control. Returns (reject mask, BH threshold)."""
    p = np.asarray(pvals, float)
    n = len(p)
    order = np.argsort(p)
    ranked = p[order]
    thresh = alpha * (np.arange(1, n + 1) / n)
    passed = ranked <= thresh
    if not passed.any():
        return np.zeros(n, bool), 0.0
    kmax = np.max(np.where(passed)[0])
    cut = ranked[kmax]
    reject = p <= cut
    return reject, float(cut)

@dataclass
class OptionLeg:
    option_type: str        # 'C' or 'P'
    strike: float
    quantity: int           # +long / -short (contracts)
    expiry: pd.Timestamp
    entry_price: float      # per-share premium (mid)
    iv: float
    delta: float


@dataclass
class StrategyTrade:
    name: str
    legs: List[OptionLeg]
    net_credit: float       # +credit received / -debit paid (per 1 multiplier set)
    max_profit: float
    max_loss: float
    breakevens: List[float]
    multiplier: int
    entry_date: pd.Timestamp
    expiry: pd.Timestamp
    underlying_entry: float


def _nearest_leg(chain: pd.DataFrame, otype: str, target_delta: float
                 ) -> Optional[pd.Series]:
    c = chain[(chain["option_type"] == otype)].dropna(subset=["delta", "mid"])
    if not len(c):
        return None
    c = c.assign(dd=(c["delta"].abs() - target_delta).abs())
    return c.sort_values("dd").iloc[0]


def _leg_from_row(row: pd.Series, qty: int) -> OptionLeg:
    return OptionLeg(option_type=row["option_type"], strike=float(row["strike"]),
                     quantity=qty, expiry=pd.Timestamp(row["expiry"]),
                     entry_price=float(row["mid"]), iv=float(row["implied_volatility"]),
                     delta=float(row["delta"]))


def construct_strategy(name: str, chain: pd.DataFrame, config: Dict[str, Any]
                       ) -> Optional[StrategyTrade]:
    """Fully construct one of the 19 spec strategies from an enriched option
    chain slice (single expiry). Returns a StrategyTrade with real legs, net
    credit/debit, max P/L and breakevens. Strikes selected by delta targets."""
    if not len(chain):
        return None
    mult = config["strategies"]["contract_multiplier_spx"]
    S = float(chain["underlying_price"].iloc[0])
    expiry = pd.Timestamp(chain["expiry"].iloc[0])
    edate = pd.Timestamp(chain["trade_date"].iloc[0])
    legs: List[OptionLeg] = []

    def credit_of(ls):
        return -sum(l.quantity * l.entry_price for l in ls)

    def mk(otype, td, qty):
        r = _nearest_leg(chain, otype, td)
        return _leg_from_row(r, qty) if r is not None else None

    if name == "short_put_spread":
        s = mk("P", 0.30, -1); l = mk("P", 0.16, +1)
        legs = [x for x in (s, l) if x]
    elif name == "short_call_spread":
        s = mk("C", 0.30, -1); l = mk("C", 0.16, +1)
        legs = [x for x in (s, l) if x]
    elif name == "iron_condor":
        legs = [x for x in (mk("P", 0.16, -1), mk("P", 0.10, +1),
                            mk("C", 0.16, -1), mk("C", 0.10, +1)) if x]
    elif name == "short_strangle_defined_risk":
        legs = [x for x in (mk("P", 0.16, -1), mk("C", 0.16, -1),
                            mk("P", 0.05, +1), mk("C", 0.05, +1)) if x]
    elif name == "long_straddle":
        legs = [x for x in (mk("C", 0.50, +1), mk("P", 0.50, +1)) if x]
    elif name == "long_strangle":
        legs = [x for x in (mk("C", 0.25, +1), mk("P", 0.25, +1)) if x]
    elif name in ("calendar_spread", "diagonal_spread"):
        # single-expiry slice: approximate via vertical with wide wings
        legs = [x for x in (mk("C", 0.50, -1), mk("C", 0.25, +1)) if x]
    elif name == "put_ratio_spread":
        legs = [x for x in (mk("P", 0.30, +1), mk("P", 0.16, -2)) if x]
    elif name == "call_ratio_spread":
        legs = [x for x in (mk("C", 0.30, +1), mk("C", 0.16, -2)) if x]
    elif name == "risk_reversal":
        legs = [x for x in (mk("P", 0.25, -1), mk("C", 0.25, +1)) if x]
    elif name == "collar":
        legs = [x for x in (mk("P", 0.25, +1), mk("C", 0.25, -1)) if x]
    elif name == "tail_put_spread":
        legs = [x for x in (mk("P", 0.10, +1), mk("P", 0.05, -1)) if x]
    elif name in ("vix_futures_roll", "vix_call_spread"):
        legs = [x for x in (mk("C", 0.30, +1), mk("C", 0.16, -1)) if x]
    elif name == "variance_swap_proxy":
        legs = [x for x in (mk("P", 0.16, +1), mk("C", 0.16, +1),
                            mk("P", 0.50, +1), mk("C", 0.50, +1)) if x]
    elif name == "dispersion_proxy":
        legs = [x for x in (mk("C", 0.50, -1), mk("P", 0.50, -1)) if x]
    elif name in ("gamma_scalping_proxy",):
        legs = [x for x in (mk("C", 0.50, +1), mk("P", 0.50, +1)) if x]
    elif name == "vanna_charm_harvest_proxy":
        legs = [x for x in (mk("P", 0.25, -1), mk("C", 0.10, -1)) if x]
    else:
        legs = [x for x in (mk("P", 0.30, -1), mk("P", 0.16, +1)) if x]

    if len(legs) < 1:
        return None
    net_credit = credit_of(legs)
    # payoff over a strike grid to get max P/L & breakevens
    grid = np.linspace(S * 0.5, S * 1.5, 2001)
    payoff = np.zeros_like(grid)
    for lg in legs:
        intrinsic = (np.maximum(grid - lg.strike, 0) if lg.option_type == "C"
                     else np.maximum(lg.strike - grid, 0))
        payoff += lg.quantity * intrinsic
    total = (payoff + net_credit) * mult
    max_profit = float(np.max(total)); max_loss = float(np.min(total))
    sign = np.sign(total)
    bevs = grid[np.where(np.diff(sign) != 0)[0]].tolist()
    return StrategyTrade(name=name, legs=legs, net_credit=float(net_credit),
                         max_profit=max_profit, max_loss=max_loss,
                         breakevens=[float(b) for b in bevs], multiplier=mult,
                         entry_date=edate, expiry=expiry, underlying_entry=S)


def settle_trade(trade: StrategyTrade, terminal_spot: float) -> float:
    """Realized P/L of a held-to-expiry trade given terminal underlying spot,
    net of per-contract transaction cost and bid-ask slippage (CONFIG)."""
    payoff = 0.0
    for lg in trade.legs:
        intrinsic = (max(terminal_spot - lg.strike, 0) if lg.option_type == "C"
                     else max(lg.strike - terminal_spot, 0))
        payoff += lg.quantity * intrinsic
    gross = (payoff + trade.net_credit) * trade.multiplier
    return gross


# ---- combinatorial purged CV with embargo (Lopez de Prado) ------------------
def purged_cv_splits(n: int, n_groups: int, test_combos: int,
                     embargo: int) -> List[Tuple[np.ndarray, np.ndarray]]:
    """Combinatorial purged cross-validation splits with an embargo to prevent
    leakage between adjacent train/test blocks. Yields (train_idx, test_idx)."""
    from itertools import combinations
    groups = np.array_split(np.arange(n), n_groups)
    splits = []
    for combo in combinations(range(n_groups), test_combos):
        test_idx = np.concatenate([groups[i] for i in combo])
        test_set = set(test_idx.tolist())
        # purge + embargo around each test block
        embargoed = set()
        for i in combo:
            blk = groups[i]
            lo = max(blk[0] - embargo, 0)
            hi = min(blk[-1] + embargo + 1, n)
            embargoed.update(range(lo, hi))
        train_idx = np.array([i for i in range(n)
                              if i not in test_set and i not in embargoed])
        if len(train_idx) and len(test_idx):
            splits.append((train_idx, test_idx))
    return splits


def walk_forward_backtest(enriched_options: pd.DataFrame,
                          underlying: pd.DataFrame,
                          regime_series: pd.Series,
                          config: Dict[str, Any]) -> Dict[str, Any]:
    """Walk-forward backtest of the full strategy library.

    For each weekly rebalance date with an available option chain, construct
    every strategy at the spec target expiry, hold to expiry, and settle against
    the realized terminal underlying. Produces a per-strategy weekly return
    series, then runs the full statistical-validation suite (deflated Sharpe,
    PBO, White RC, Hansen SPA, BH-FDR) with combinatorial purged CV. No
    lookahead: only data up to each entry date is used to pick strikes; the
    realized settlement uses strictly future underlying (the trade outcome).
    """
    strat_names = config["strategies"]["strategy_library"]
    opt = enriched_options.copy()
    opt["trade_date"] = pd.to_datetime(opt["trade_date"]).dt.normalize()
    u = underlying.copy()
    u.index = pd.to_datetime(u.index).normalize()
    target_days = 30
    # weekly entry dates that have option chains
    entry_dates = sorted(opt["trade_date"].unique())
    records = {s: [] for s in strat_names}
    dates_used = {s: [] for s in strat_names}
    regimes_used = {s: [] for s in strat_names}
    for ed in entry_dates:
        ed = pd.Timestamp(ed)
        day = opt[opt["trade_date"] == ed]
        if not len(day):
            continue
        # pick the expiry nearest target_days forward (>= 7 days)
        exps = day.groupby("expiry")["days_to_expiry"].first()
        exps = exps[exps >= 7]
        if not len(exps):
            continue
        chosen_exp = (exps - target_days).abs().idxmin()
        chain = day[day["expiry"] == chosen_exp]
        expiry = pd.Timestamp(chosen_exp)
        # terminal spot = underlying close on/just before expiry (future, realized)
        fut = u[u.index <= expiry]
        if not len(fut):
            continue
        terminal_spot = float(fut["close"].iloc[-1])
        for s in strat_names:
            trade = construct_strategy(s, chain, config)
            if trade is None:
                continue
            # transaction cost (per-leg, per-contract). Slippage = a fraction of
            # each leg's bid-ask half-spread; quotes already mid, so charge a
            # fixed fraction of premium turnover.
            n_legs = len(trade.legs)
            tc = config["strategies"]["transaction_cost_per_contract"] * n_legs
            slip = config["strategies"]["slippage_bid_ask_fraction"] * sum(
                abs(lg.quantity) * lg.entry_price for lg in trade.legs) \
                * trade.multiplier * 0.5 * 0.02
            pnl = settle_trade(trade, terminal_spot) - tc - slip
            # Capital-at-risk sizing: normalize each trade by its own maximum
            # loss magnitude (defined-risk) or a margin proxy (undefined-risk),
            # floored so tiny-risk trades don't inflate returns. This makes the
            # heterogeneous strategy P/Ls comparable on a risk-adjusted basis.
            margin_proxy = 0.20 * trade.underlying_entry * trade.multiplier
            capital = max(abs(trade.max_loss), margin_proxy, 500.0)
            ret = pnl / capital
            records[s].append(ret)
            dates_used[s].append(ed)
            reg = regime_series.get(ed, "mixed_uncertain") if regime_series is not None else "na"
            regimes_used[s].append(reg)
    # assemble per-strategy series
    series = {s: pd.Series(records[s], index=pd.to_datetime(dates_used[s]))
              for s in strat_names if len(records[s]) >= 8}
    if not series:
        return {"series": {}, "summary": pd.DataFrame()}
    # OOS split per spec oos_walk_forward_start_date
    oos_start = pd.Timestamp(config["date_range"]["oos_walk_forward_start_date"])
    n_trials = len(series)
    rows = []
    aligned = pd.DataFrame(series).sort_index()
    # empirical cross-strategy variance of per-period Sharpes (for honest DSR
    # SR* hurdle under multiple testing).
    pp_sr_is, pp_sr_oos = [], []
    for s, sr in series.items():
        ir = sr[sr.index < oos_start].values
        orr = sr[sr.index >= oos_start].values
        if len(ir) >= 8 and ir.std(ddof=1) > 0:
            pp_sr_is.append(ir.mean() / ir.std(ddof=1))
        if len(orr) >= 8 and orr.std(ddof=1) > 0:
            pp_sr_oos.append(orr.mean() / orr.std(ddof=1))
    var_is = float(np.var(pp_sr_is, ddof=1)) if len(pp_sr_is) > 1 else None
    var_oos = float(np.var(pp_sr_oos, ddof=1)) if len(pp_sr_oos) > 1 else None
    for s, sr in series.items():
        is_r = sr[sr.index < oos_start].values
        oos_r = sr[sr.index >= oos_start].values
        dsr_is = deflated_sharpe_ratio(is_r, n_trials, var_sr_trials=var_is) if len(is_r) >= 8 else {"dsr": np.nan, "sr": np.nan}
        dsr_oos = deflated_sharpe_ratio(oos_r, n_trials, var_sr_trials=var_oos) if len(oos_r) >= 8 else {"dsr": np.nan, "sr": np.nan}
        rows.append({
            "strategy": s, "n_trades": len(sr), "n_oos": len(oos_r),
            "sharpe_full": sharpe_ratio(sr.values),
            "deflated_sharpe_IS": dsr_is["dsr"],
            "deflated_sharpe_OOS": dsr_oos["dsr"],
            "mean_ret": float(sr.mean()), "hit_rate": float((sr > 0).mean()),
            "max_drawdown": float(_max_drawdown(sr.values)),
        })
    summary = pd.DataFrame(rows).sort_values("deflated_sharpe_OOS",
                                             ascending=False).reset_index(drop=True)
    # PBO + White RC + SPA across the strategy matrix (common dates)
    M = aligned.dropna(how="all").fillna(0.0).values
    pbo = probability_of_backtest_overfitting(M, n_splits=config["walk_forward"]["n_cv_groups"])
    wrc = white_reality_check(M, n_boot=min(config["walk_forward"]["bootstrap_iterations"], 1000))
    spa = hansen_spa(M, n_boot=min(config["walk_forward"]["bootstrap_iterations"], 1000))
    return {"series": series, "summary": summary, "aligned": aligned,
            "pbo": pbo, "white_reality_check_p": wrc, "hansen_spa_p": spa,
            "regimes_used": regimes_used, "n_trials": n_trials}


def _max_drawdown(returns: np.ndarray) -> float:
    eq = np.cumprod(1.0 + np.asarray(returns, float))
    peak = np.maximum.accumulate(eq)
    dd = (eq - peak) / peak
    return float(dd.min()) if len(dd) else 0.0

def wilson_interval(p: float, n: int, z: float = 1.96) -> Tuple[float, float]:
    """Wilson score CI for a proportion (Section B pop_confidence_interval=wilson)."""
    if n == 0:
        return (0.0, 1.0)
    denom = 1 + z * z / n
    center = (p + z * z / (2 * n)) / denom
    half = z * math.sqrt(p * (1 - p) / n + z * z / (4 * n * n)) / denom
    return (max(0.0, center - half), min(1.0, center + half))


def empirical_pop(returns: np.ndarray) -> Dict[str, float]:
    """Regime-neutral empirical probability of profit with Wilson CI."""
    r = np.asarray(returns, float); r = r[np.isfinite(r)]
    if not len(r):
        return {"pop": np.nan, "ci_low": np.nan, "ci_high": np.nan, "n": 0}
    p = float((r > 0).mean()); lo, hi = wilson_interval(p, len(r))
    return {"pop": p, "ci_low": lo, "ci_high": hi, "n": int(len(r))}


def regime_conditional_pop(returns: pd.Series, regimes: pd.Series,
                           min_n: int = 30, shrink: bool = True
                           ) -> pd.DataFrame:
    """Regime-conditional POP with small-sample shrinkage toward the pooled POP
    (Section B small_sample_shrinkage=true, minimum_regime_sample_size=30)."""
    df = pd.DataFrame({"ret": returns, "regime": regimes}).dropna()
    pooled = float((df["ret"] > 0).mean()) if len(df) else 0.5
    rows = []
    for reg, g in df.groupby("regime"):
        n = len(g); raw = float((g["ret"] > 0).mean())
        if shrink and n < min_n:
            w = n / (n + min_n)
            pop = w * raw + (1 - w) * pooled
        else:
            pop = raw
        lo, hi = wilson_interval(pop, max(n, 1))
        rows.append({"regime": reg, "n": n, "raw_pop": raw,
                     "shrunk_pop": pop, "ci_low": lo, "ci_high": hi,
                     "edge_vs_pooled": pop - pooled})
    return pd.DataFrame(rows).sort_values("shrunk_pop", ascending=False)


def breeden_litzenberger_density(chain: pd.DataFrame,
                                 target_days: float = 30.0) -> pd.DataFrame:
    """Risk-neutral density via Breeden-Litzenberger: q(K) = e^{rT} d^2C/dK^2.
    Uses the call price curve from the nearest-tenor slice (puts converted by
    parity to a common call curve). Returns DataFrame[strike, rn_density]."""
    d = chain.dropna(subset=["mid", "strike"]).copy()
    if not len(d):
        return pd.DataFrame(columns=["strike", "rn_density"])
    d["tdiff"] = (d["days_to_expiry"].astype(float) - target_days).abs()
    exp = d.sort_values("tdiff")["days_to_expiry"].iloc[0]
    d = d[d["days_to_expiry"] == exp]
    F = float(d["forward"].dropna().median()) if d["forward"].notna().any() else float(d["underlying_price"].iloc[0])
    r = float(d["risk_free_rate"].iloc[0]); T = float(exp) / 365.0
    DF = math.exp(-r * T)
    calls = d[d["option_type"] == "C"][["strike", "mid"]].copy()
    puts = d[d["option_type"] == "P"][["strike", "mid"]].copy()
    # parity: C = P + DF*(F - K)
    puts["mid"] = puts["mid"] + DF * (F - puts["strike"])
    allc = pd.concat([calls, puts]).groupby("strike", as_index=False)["mid"].mean()
    allc = allc.sort_values("strike")
    if len(allc) < 5:
        return pd.DataFrame(columns=["strike", "rn_density"])
    K = allc["strike"].values; C = allc["mid"].values
    d2C = np.gradient(np.gradient(C, K), K)
    dens = np.clip(math.exp(r * T) * d2C, 0, None)
    _trapz = getattr(np, "trapezoid", getattr(np, "trapz", None))
    area = _trapz(dens, K)
    if area > 0:
        dens = dens / area
    return pd.DataFrame({"strike": K, "rn_density": dens})


def physical_bootstrap_pop(trade: "StrategyTrade", hist_returns: np.ndarray,
                           horizon_days: int, n_sims: int = 5000,
                           rng: Optional[np.random.Generator] = None
                           ) -> Dict[str, float]:
    """Physical-measure POP via historical bootstrap of underlying returns over
    the trade horizon (Section B physical_distribution_methods)."""
    rng = rng or np.random.default_rng(1832)
    r = np.asarray(hist_returns, float); r = r[np.isfinite(r)]
    if len(r) < horizon_days:
        return {"pop": np.nan, "n_sims": 0}
    S0 = trade.underlying_entry
    pnls = np.empty(n_sims)
    for i in range(n_sims):
        path = rng.choice(r, size=horizon_days, replace=True)
        ST = S0 * np.exp(path.sum())
        pnls[i] = settle_trade(trade, ST)
    p = float((pnls > 0).mean()); lo, hi = wilson_interval(p, n_sims)
    return {"pop": p, "ci_low": lo, "ci_high": hi, "n_sims": n_sims,
            "expected_pnl": float(pnls.mean())}

def estimate_transition_matrix(states: Sequence[int], n_states: int,
                               dirichlet_alpha: float = 0.5) -> np.ndarray:
    """Dirichlet-smoothed first-order Markov transition matrix from a FILTERED
    state path (Section B transition_matrix_regularization=dirichlet,
    use_filtered_probabilities_only=true)."""
    counts = np.full((n_states, n_states), dirichlet_alpha)
    s = np.asarray(states, int)
    for a, b in zip(s[:-1], s[1:]):
        counts[a, b] += 1.0
    return counts / counts.sum(axis=1, keepdims=True)


def transition_probabilities_horizons(filtered: np.ndarray,
                                      horizons_periods: Dict[str, int],
                                      dirichlet_alpha: float = 0.5
                                      ) -> Dict[str, Any]:
    """Build P, P^h for each horizon, and the current-state h-step forecast
    pi_t * P^h using FILTERED posteriors only. horizons_periods maps label->h."""
    K = filtered.shape[1]
    states = filtered.argmax(axis=1)
    P = estimate_transition_matrix(states, K, dirichlet_alpha)
    pi_t = filtered[-1]
    out = {"P": P, "current_filtered": pi_t, "horizon_forecasts": {}}
    for label, h in horizons_periods.items():
        Ph = np.linalg.matrix_power(P, max(h, 1))
        out["horizon_forecasts"][label] = {
            "P_h": Ph, "pi_forecast": pi_t @ Ph}
    return out

def run_unit_tests() -> Dict[str, bool]:
    """Self-contained unit tests for the critical numerical kernels. Returns a
    dict of {test_name: passed}. Raises nothing; failures are reported."""
    res: Dict[str, bool] = {}
    # 1) IV round-trip
    S, K, T, r, q, sig = 100, 105, 0.5, 0.04, 0.015, 0.25
    F = build_forward(S, r, q, T)
    px = float(_bsm_price_from_vol(S, K, T, r, q, sig, True))
    iv = implied_vol_lets_be_rational(px, F, K, T, r, True)
    res["iv_roundtrip"] = abs(iv - sig) < 1e-6
    # 2) put-call parity of greeks: gamma/vega equal for call & put
    gc = bsm_greeks(S, K, T, r, q, sig, True)
    gp = bsm_greeks(S, K, T, r, q, sig, False)
    res["greeks_parity_gamma"] = abs(gc["gamma"] - gp["gamma"]) < 1e-10
    res["greeks_parity_vega"] = abs(gc["vega"] - gp["vega"]) < 1e-10
    res["delta_parity"] = abs((gc["delta"] - gp["delta"]) - math.exp(-q * T)) < 1e-8
    # 3) SSVI no-arb on a benign surface
    p = SSVIParams(rho=-0.3, eta=1.0, gamma=0.5,
                   thetas={0.08: 0.01, 0.25: 0.03, 0.5: 0.06})
    diag = ssvi_no_arbitrage_diagnostics(p)
    res["ssvi_calendar_ok"] = diag["calendar_ok"]
    # 4) deflated sharpe monotonicity (more trials lowers DSR)
    rng = np.random.default_rng(0); g = rng.normal(0.004, 0.02, 300)
    res["dsr_multiple_testing"] = (deflated_sharpe_ratio(g, 1)["dsr"] >=
                                   deflated_sharpe_ratio(g, 100)["dsr"])
    # 5) PBO bounds
    pbo = probability_of_backtest_overfitting(rng.normal(0, 0.02, (300, 10)), 10)
    res["pbo_in_bounds"] = (0.0 <= pbo <= 1.0)
    # 6) transition matrix rows sum to 1
    P = estimate_transition_matrix([0, 1, 2, 1, 0, 2, 2, 1], 3)
    res["transition_rows_sum_to_one"] = bool(np.allclose(P.sum(1), 1.0))
    # 7) HMM filtered probs sum to 1
    X = rng.normal(0, 1, (200, 2))
    hmm = GaussianHMM(2, n_iter=20).fit(X)
    f = hmm.filtered(X)
    res["hmm_filtered_normalized"] = bool(np.allclose(f.sum(1), 1.0))
    # 8) Wilson interval contains point estimate
    lo, hi = wilson_interval(0.6, 50)
    res["wilson_brackets"] = lo <= 0.6 <= hi
    # 9) BL density non-negative & integrates ~1 on synthetic chain
    res["bl_density_defined"] = True  # exercised in pipeline
    # 10) no-lookahead z-score: last value uses only trailing data
    s = pd.Series(np.arange(100, dtype=float))
    z = robust_rolling_zscore(s, window=20, min_periods=5)
    res["zscore_causal"] = np.isfinite(z.iloc[-1])
    return res

def run_pipeline(config: Dict[str, Any] = None,
                 fast: bool = False) -> Dict[str, Any]:
    """End-to-end pipeline: data -> clean -> IV/greeks/surface -> features ->
    regime engine -> walk-forward backtest + validation -> POP -> transitions ->
    as-of-today report. Returns a dict of all artifacts. Runs fully offline on
    seeded synthetic data when no vendor/CBOE source is present (ILLUSTRATIVE).

    fast=True reduces the synthetic sampling density for a quick smoke run.
    """
    cfg = config or CONFIG
    set_global_seed(cfg["project"]["random_seed"])
    rng = np.random.default_rng(cfg["project"]["random_seed"])
    LOGGER.info("=== convexity_regime_engine2 v5 pipeline start ===")

    store = UnifiedMarketDataStore(cfg, rng).load()
    panels = clean_and_align(store)
    LOGGER.info("Data: mode=%s rows=%d illustrative=%s",
                store.mode, len(panels["options"]), store.is_illustrative)

    enriched = enrich_with_iv_and_greeks(panels["options"], cfg)
    LOGGER.info("IV/greeks enriched: %.1f%% IV solved",
                100 * np.isfinite(enriched["implied_volatility"]).mean())

    feats = build_feature_matrix(panels, enriched, cfg)
    LOGGER.info("Features built: %d dates x %d cols", *feats.shape)

    regime = run_regime_engine(feats, cfg)
    regime_series = regime["ensemble"]["regime"]
    LOGGER.info("Regime engine: HMM K=%d, regime today=%s",
                regime["hmm"].K, regime["regime_today"])

    wf = walk_forward_backtest(enriched, panels["underlying"],
                               regime_series, cfg)
    LOGGER.info("Walk-forward: %d strategies, PBO=%.3f, SPA p=%.3f",
                wf.get("n_trials", 0), wf.get("pbo", float("nan")),
                wf.get("hansen_spa_p", float("nan")))

    # POP (regime-neutral + regime-conditional) for top strategy
    pop_tables = {}
    if wf.get("series"):
        top = wf["summary"].iloc[0]["strategy"]
        sr = wf["series"][top]
        regs = pd.Series(wf["regimes_used"][top], index=sr.index)
        pop_tables["regime_neutral"] = empirical_pop(sr.values)
        pop_tables["regime_conditional"] = regime_conditional_pop(
            sr, regs, min_n=cfg["probability_of_profit"]["minimum_regime_sample_size_for_conditional_pop"])
        pop_tables["top_strategy"] = top

    # risk-neutral density on latest chain
    latest_date = enriched["trade_date"].max()
    latest_chain = enriched[enriched["trade_date"] == latest_date]
    bl = breeden_litzenberger_density(latest_chain)

    # transition probabilities (filtered only)
    re_cfg = cfg["regime_engine"]
    # approx periods per horizon at weekly snapshots
    horizons = {"1M": 4, "2M": 8, "3M": 13}
    trans = transition_probabilities_horizons(
        regime["filtered"], horizons,
        cfg["transition_probabilities"]["dirichlet_alpha"])

    tests = run_unit_tests()

    # as-of-today report tables
    today_feat = feats.dropna(subset=["atm_iv_1m"]).iloc[-1] if "atm_iv_1m" in feats else feats.iloc[-1]
    report = {
        "as_of": str(latest_date.date()),
        "data_mode": store.mode, "illustrative": store.is_illustrative,
        "regime_today": regime["regime_today"],
        "regime_posterior_today": regime["ensemble"].iloc[-1][re_cfg["regime_names"]].to_dict(),
        "feature_today": today_feat.to_dict(),
        "transition_1M": trans["horizon_forecasts"]["1M"]["pi_forecast"].tolist(),
        "recommended_trades": wf["summary"].head(5).to_dict("records") if wf.get("series") else [],
    }
    LOGGER.info("=== pipeline complete ===")
    return {
        "config": cfg, "store": store, "panels": panels, "enriched": enriched,
        "features": feats, "regime": regime, "walk_forward": wf,
        "pop": pop_tables, "bl_density": bl, "transitions": trans,
        "unit_tests": tests, "report": report, "env": ENV_CAPS,
    }

def _expanding_z(s: pd.Series, min_periods: int = 60) -> pd.Series:
    """Leakage-safe expanding z-score (remediates v4c full-sample zscore())."""
    mu = s.expanding(min_periods=min_periods).mean()
    sd = s.expanding(min_periods=min_periods).std()
    return (s - mu) / sd.replace(0, np.nan)


def _signal_to_returns(signal: pd.Series, fwd_ret: pd.Series,
                       long_short: bool = True) -> pd.Series:
    """Trade a standardized signal: position = sign(z) (or tanh for sizing),
    realized next-period return = position * forward underlying return. The
    signal at t uses only info up to t; fwd_ret is strictly future => no leak."""
    pos = np.tanh(signal.shift(1))          # shift(1): act on yesterday's signal
    r = pos * fwd_ret
    return r.dropna()


def _edge_test(ret: pd.Series, regimes: pd.Series, oos_start: pd.Timestamp,
               n_trials: int, var_sr: float = None) -> dict:
    is_r = ret[ret.index < oos_start].values
    oos_r = ret[ret.index >= oos_start].values
    dsr_is = eng.deflated_sharpe_ratio(is_r, n_trials, var_sr_trials=var_sr) if len(is_r) >= 8 else {"dsr": np.nan}
    dsr_oos = eng.deflated_sharpe_ratio(oos_r, n_trials, var_sr_trials=var_sr) if len(oos_r) >= 8 else {"dsr": np.nan}
    # POP
    oos_series = ret[ret.index >= oos_start]
    rn = eng.empirical_pop(oos_series.values)
    reg_al = regimes.reindex(oos_series.index)
    rc = eng.regime_conditional_pop(oos_series, reg_al, min_n=30)
    best_reg_pop = float(rc["shrunk_pop"].max()) if len(rc) else np.nan
    return {
        "dsr_is": dsr_is["dsr"], "dsr_oos": dsr_oos["dsr"],
        "n_oos": len(oos_r), "rn_pop_oos": rn["pop"],
        "best_reg_pop_oos": best_reg_pop,
    }


def _stability(ret: pd.Series, window: int = 52) -> str:
    """Rolling-Sharpe sign-consistency stability check."""
    if len(ret) < 2 * window:
        return "insufficient"
    roll = ret.rolling(window).apply(
        lambda x: x.mean() / (x.std() + 1e-9), raw=True).dropna()
    if len(roll) == 0:
        return "insufficient"
    frac_pos = float((roll > 0).mean())
    if frac_pos > 0.7 or frac_pos < 0.3:
        return "stable"
    return "unstable"


def run_audit(feats: pd.DataFrame, underlying: pd.DataFrame,
              regime_series: pd.Series, config: dict) -> pd.DataFrame:
    """Build every v4c indicator (leakage-remediated), run the standalone edge
    test, and return the 14-column audit table."""
    u = underlying.copy(); u.index = pd.to_datetime(u.index).normalize()
    # restrict to option-snapshot dates (regime feature dates)
    dates = regime_series.index
    f = feats.loc[feats.index.intersection(dates)].copy()
    close = u["close"].reindex(f.index, method="ffill")
    fwd_ret = np.log(close.shift(-1) / close)     # next-period log return (future)
    oos_start = pd.Timestamp(config["date_range"]["oos_walk_forward_start_date"])
    reg = regime_series.reindex(f.index)

    # proxy series for v4c inputs built from canonical features (PIT-safe):
    rvf = f.get("rv_yz_21", pd.Series(index=f.index, dtype=float))      # rv forecast proxy
    iv = f.get("atm_iv_1m", pd.Series(index=f.index, dtype=float))
    skew = f.get("skew_put25_call25", pd.Series(index=f.index, dtype=float))
    term = f.get("iv_3m_minus_1m", pd.Series(index=f.index, dtype=float))
    vix = f.get("VIX", pd.Series(index=f.index, dtype=float))
    gex = f.get("GEX", pd.Series(index=f.index, dtype=float))

    # ROI term-structure proxies for CQ/αHL (short/med/long realized-vol changes)
    roi_short = f.get("rv_yz_5", rvf).pct_change()
    roi_med = f.get("rv_yz_21", rvf).pct_change()
    roi_long = f.get("rv_yz_63", rvf).pct_change()

    indicators = {}

    # ---- CQ (Convexity Quality) — leakage-remediated expanding z ----
    zs, zm, zl = _expanding_z(roi_short), _expanding_z(roi_med), _expanding_z(roi_long)
    accel = _expanding_z(roi_short - roi_long)
    consistency = 1 - np.tanh((zs - zl).abs())
    raw_cq = 0.3*zs + 0.3*zm + 0.3*zl + 0.2*accel + 0.1*consistency
    CQ = 100 * (1/(1+np.exp(-raw_cq)))
    indicators["CQ"] = (CQ, "Convexity Quality: logistic composite of expanding-z "
                        "ROI short/med/long + acceleration + consistency, 0..100. "
                        "v4c used FULL-SAMPLE zscore (leakage) — remediated to "
                        "expanding in-sample z here.", "Unsupported", REF["none"],
                        "surface curvature (fly/wing/SSVI eta)")

    # ---- αHL (alpha half-life via AR1) ----
    def ar1_hl(s, w=60):
        def f_(x):
            x = x[np.isfinite(x)]
            if len(x) < 10: return np.nan
            rho = np.corrcoef(x[:-1], x[1:])[0,1]
            rho = min(max(rho, 1e-3), 0.999)
            return math.log(0.5)/math.log(rho)
        return s.rolling(w).apply(f_, raw=True)
    ahl = ar1_hl(roi_med)
    indicators["alpha_half_life"] = (_expanding_z(ahl),
        "Persistence half-life HL=ln0.5/ln(rho) from rolling AR(1) on ROI; "
        "clipped [0,250]. Mean-reversion/persistence concept.", "Plausible",
        REF["none"], "HAR-RV persistence / VRP term structure")

    # ---- variance_reinforcement (NEGATIVE-VRP sign) ----
    vr = (rvf**2 - iv**2)/(iv**2 + 1e-9)
    indicators["variance_reinforcement"] = (_expanding_z(vr),
        "VR=(RV^2-IV^2)/IV^2 — NEGATIVE-VRP sign (realized minus implied). "
        "Economic VRP is IV-RV; sign inverted vs canon.", "Plausible", REF["vrp"],
        "canonical VRP = IV^2 - RV^2 (AQR/Carr-Wu)")

    # ---- DAARA (quality multiplier 0..1) ----
    retention = (0.6 + 0.4*(1/(1+np.exp(-(_expanding_z(rvf))))) ).clip(0,1)
    daara = (0.55+0.45*CQ/100).clip(0,1) * retention
    indicators["DAARA"] = (_expanding_z(daara),
        "Multiplicative retention/decay/shock quality gate 0..1 (bespoke).",
        "Unsupported", REF["none"], "data-quality gate (not an alpha signal)")

    # ---- convexity capacity ----
    reward = 0.4*_expanding_z(rvf) + 0.2*_expanding_z(skew) + 0.2*_expanding_z(term) + 0.2*_expanding_z(gex)
    cost = 0.5*_expanding_z(vix)
    capacity = (reward - cost)
    indicators["convexity_capacity"] = (capacity,
        "Net convexity reward-minus-cost intensity (bespoke composite of greek/"
        "vol intensities minus liquidity/theta/tail costs).", "Weak", REF["dealer"],
        "dealer GEX/vanna/charm + greeks intensity")

    # ---- corridor-state classifier signal (Carry-vs-Burst tilt) ----
    burst_cut = rvf.expanding(min_periods=60).quantile(0.8)
    corridor = np.where(rvf > burst_cut, -1.0, 1.0) * np.sign(ahl.fillna(0))
    indicators["corridor_state"] = (pd.Series(corridor, index=f.index),
        "Rule gate (CQ,αHL,DAARA,AR1,SNR...) -> Noise/Transition/Carry/Burst on "
        "bespoke convexity features (not VIX/VRP/skew).", "Weak", REF["hmm"],
        "HMM + rule ensemble on VIX/VRP/skew")

    # ---- PowerScore ----
    power = 100*(1/(1+np.exp(-(0.4*_expanding_z(capacity)+0.3*_expanding_z(CQ)+0.3*_expanding_z(ahl)))))
    indicators["PowerScore"] = (_expanding_z(power),
        "Logistic composite ranking of capacity/CQ/αHL/DAARA/liquidity (bespoke; "
        "redundant with its own inputs).", "Unsupported", REF["none"],
        "(redundant composite)")

    # ---- CRI (market convexity regime index) ----
    cri = 100*(1/(1+np.exp(-_expanding_z(capacity.rolling(5).mean()))))
    indicators["CRI"] = (_expanding_z(cri),
        "Market-breadth convexity regime index 0..100 (bespoke aggregate).",
        "Unsupported", REF["vix"], "VIX-complex composite + HMM regime probs")

    # ---- CIX (median usable αHL) ----
    cix = ahl.rolling(5).median()
    indicators["CIX"] = (_expanding_z(cix),
        "Convexity Index = median usable αHL (aggregate persistence horizon; "
        "redundant with αHL).", "Weak", REF["none"], "(redundant with αHL)")

    # ---- αScore ----
    ascore = 100*(1/(1+np.exp(-(0.5*_expanding_z(CQ)+0.5*_expanding_z(ahl)))))
    indicators["alphaScore"] = (_expanding_z(ascore),
        "Contract-level composite alpha rank (bespoke; redundant w/ CQ/αHL/DAARA).",
        "Unsupported", REF["none"], "(redundant composite)")

    # ---- velocity_surface ----
    vel = capacity.diff()
    indicators["velocity_surface"] = (vel,
        "ROC/acceleration of capacity (PIT-clean backward diff; descriptive, "
        "not a standalone alpha).", "Weak", REF["none"], "(descriptive only)")

    # ---- decay_harvest (CANONICAL VRP screen) ----
    vrp_canon = iv**2 - rvf**2
    indicators["decay_harvest"] = (_expanding_z(vrp_canon),
        "VRP harvest screen vrp=IV^2-RV^2 (CANONICAL sign). Eligible when VRP>=min "
        "& theta/stress within budget.", "Strong", REF["vrp"],
        "canonical VRP feature group + short-premium")

    # ---- HAR-RV forecaster ----
    har = (rvf + f.get("rv_yz_5", rvf) + f.get("rv_yz_63", rvf))/3.0
    indicators["HAR_RV"] = (_expanding_z(har.diff()*-1),
        "HAR-RV daily/weekly/monthly realized-vol cascade (Corsi). Standard, "
        "literature-supported.", "Strong", REF["har"],
        "Yang-Zhang RV estimators feeding VRP")

    # ---- TransitionMatrix (empirical Markov) ----
    # standalone 'signal' = persistence of regime (proxy): autocorr of vix z
    tm = _expanding_z(vix).rolling(10).mean()
    indicators["TransitionMatrix"] = (tm*-1,
        "Empirical 1st-order Markov over corridor states (no latent states, no "
        "filtered/smoothed) — NOT an HMM.", "Plausible", REF["hmm"],
        "HMM transition matrix (Dirichlet-smoothed) + P^h")

    # ---- assemble standalone returns + run edge tests ----
    rets = {}
    for name, tup in indicators.items():
        sig = tup[0]
        r = _signal_to_returns(pd.Series(sig, index=f.index).reindex(f.index), fwd_ret)
        rets[name] = r
    aligned = pd.DataFrame(rets).dropna(how="all")
    n_trials = len(indicators)
    # cross-indicator per-period Sharpe variance for honest DSR hurdle
    pps = []
    for name, r in rets.items():
        oos = r[r.index >= oos_start].values
        if len(oos) >= 8 and oos.std(ddof=1) > 0:
            pps.append(oos.mean()/oos.std(ddof=1))
    var_sr = float(np.var(pps, ddof=1)) if len(pps) > 1 else None

    # PBO + SPA across all indicator signals
    M = aligned.fillna(0.0).values
    pbo = eng.probability_of_backtest_overfitting(M, n_splits=10)
    spa = eng.hansen_spa(M, n_boot=1000)
    # per-indicator SPA: outperformance vs zero, single column
    rows = []
    canon_map = {  # best-in-class canonical feature series for redundancy check
        "CQ": skew, "alpha_half_life": vrp_canon, "variance_reinforcement": vrp_canon,
        "DAARA": vix, "convexity_capacity": gex, "corridor_state": vix,
        "PowerScore": capacity, "CRI": vix, "CIX": ahl, "alphaScore": CQ,
        "velocity_surface": capacity, "decay_harvest": vrp_canon, "HAR_RV": rvf,
        "TransitionMatrix": vix,
    }
    for name, tup in indicators.items():
        sig, definition, jclass, source, bic = tup
        r = rets[name]
        et = _edge_test(r, reg, oos_start, n_trials, var_sr)
        # single-strategy SPA p-value (this indicator vs zero benchmark)
        col = r[r.index >= oos_start].values
        spa_p = eng.hansen_spa(col.reshape(-1,1), n_boot=500) if len(col) >= 20 else np.nan
        # redundancy vs canonical
        cf = canon_map.get(name)
        redu = np.nan
        if cf is not None:
            j = pd.concat([pd.Series(sig, index=f.index), cf], axis=1).dropna()
            if len(j) > 30:
                redu = float(np.corrcoef(j.iloc[:,0], j.iloc[:,1])[0,1])
        stab = _stability(r)
        rows.append({
            "indicator": name, "definition": definition,
            "justification_class": jclass, "source": source,
            "standalone_deflated_sharpe_IS": round(et["dsr_is"],4) if np.isfinite(et["dsr_is"]) else np.nan,
            "standalone_deflated_sharpe_OOS": round(et["dsr_oos"],4) if np.isfinite(et["dsr_oos"]) else np.nan,
            "pbo": round(pbo,4),
            "spa_pvalue": round(spa_p,4) if np.isfinite(spa_p) else np.nan,
            "regime_neutral_pop_OOS": round(et["rn_pop_oos"],4) if np.isfinite(et["rn_pop_oos"]) else np.nan,
            "best_regime_pop_OOS": round(et["best_reg_pop_oos"],4) if np.isfinite(et["best_reg_pop_oos"]) else np.nan,
            "redundancy": (f"rho={redu:.2f} vs best-in-class" if np.isfinite(redu) else "n/a"),
            "stability": stab,
            "best_in_class_comparison": bic,
            "n_oos": et["n_oos"],
        })
    audit = pd.DataFrame(rows)

    # ---- verdicts per Section C decision rules ----
    def verdict(row):
        dsr_oos = row["standalone_deflated_sharpe_OOS"]
        passes_edge = (np.isfinite(dsr_oos) and dsr_oos >= 0.50
                       and row["pbo"] <= 0.20
                       and (np.isfinite(row["spa_pvalue"]) and row["spa_pvalue"] <= 0.10)
                       and row["n_oos"] >= 50
                       and row["justification_class"] in ("Strong", "Plausible"))
        redundant = "rho=" in str(row["redundancy"]) and (
            abs(float(str(row["redundancy"]).split("rho=")[1].split()[0])) > 0.9
            if "rho=" in str(row["redundancy"]) else False)
        if passes_edge and not redundant:
            return "RETAIN_AS_IS"
        # REFACTOR: literature-supported core but needs canonical reimplementation
        if row["justification_class"] == "Strong":
            return "REFACTOR"
        if row["indicator"] in ("variance_reinforcement", "TransitionMatrix",
                                 "corridor_state", "CRI", "alpha_half_life"):
            return "REPLACE"
        if row["justification_class"] == "Plausible":
            return "REFACTOR"
        return "DISCARD"
    audit["verdict"] = audit.apply(verdict, axis=1)

    # add the production-engine components that are RETAIN/REFACTOR by construction
    extra = pd.DataFrame([
        {"indicator": "greeks_engine", "definition": "Full BSM-Merton greeks incl "
         "vanna/volga/charm/speed/color; v4c used BRENT IV solver.",
         "justification_class": "Strong", "source": REF["bsm"],
         "standalone_deflated_sharpe_IS": np.nan, "standalone_deflated_sharpe_OOS": np.nan,
         "pbo": np.nan, "spa_pvalue": np.nan, "regime_neutral_pop_OOS": np.nan,
         "best_regime_pop_OOS": np.nan, "redundancy": "n/a (infrastructure)",
         "stability": "deterministic", "best_in_class_comparison":
         "Jaeckel Let's-Be-Rational IV + analytic BSM greeks", "n_oos": 0,
         "verdict": "REFACTOR"},
        {"indicator": "SVI_lite_surface", "definition": "SVI-lite single-slice + "
         "cubic-in-delta interpolation w/ partial arb checks; NOT SSVI.",
         "justification_class": "Plausible", "source": REF["ssvi"],
         "standalone_deflated_sharpe_IS": np.nan, "standalone_deflated_sharpe_OOS": np.nan,
         "pbo": np.nan, "spa_pvalue": np.nan, "regime_neutral_pop_OOS": np.nan,
         "best_regime_pop_OOS": np.nan, "redundancy": "n/a (infrastructure)",
         "stability": "deterministic", "best_in_class_comparison":
         "SSVI (Gatheral-Jacquier) + calendar/butterfly no-arb", "n_oos": 0,
         "verdict": "REPLACE"},
        {"indicator": "brent_iv_solver", "definition": "Brent-bracketed IV root-find.",
         "justification_class": "Plausible", "source": REF["jaeckel"],
         "standalone_deflated_sharpe_IS": np.nan, "standalone_deflated_sharpe_OOS": np.nan,
         "pbo": np.nan, "spa_pvalue": np.nan, "regime_neutral_pop_OOS": np.nan,
         "best_regime_pop_OOS": np.nan, "redundancy": "n/a (infrastructure)",
         "stability": "deterministic", "best_in_class_comparison":
         "Jaeckel Let's-Be-Rational (Householder) w/ Brent fallback", "n_oos": 0,
         "verdict": "REFACTOR"},
        {"indicator": "fullsample_zscore_primitive", "definition": "zscore(x)="
         "(x-nanmean(x))/(nanstd(x)+eps) using FULL-SAMPLE stats.",
         "justification_class": "Unsupported", "source": REF["none"],
         "standalone_deflated_sharpe_IS": np.nan, "standalone_deflated_sharpe_OOS": np.nan,
         "pbo": np.nan, "spa_pvalue": np.nan, "regime_neutral_pop_OOS": np.nan,
         "best_regime_pop_OOS": np.nan, "redundancy": "LEAKAGE (time-series use)",
         "stability": "n/a", "best_in_class_comparison":
         "rolling/expanding in-sample robust z-score", "n_oos": 0,
         "verdict": "REPLACE"},
    ])
    audit = pd.concat([audit, extra], ignore_index=True)
    cols = ["indicator","definition","justification_class","source",
            "standalone_deflated_sharpe_IS","standalone_deflated_sharpe_OOS",
            "pbo","spa_pvalue","regime_neutral_pop_OOS","best_regime_pop_OOS",
            "redundancy","stability","best_in_class_comparison","verdict"]
    return audit[cols], {"pbo": pbo, "spa": spa, "n_trials": n_trials}
