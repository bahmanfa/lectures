P_filt[t] + J @ (P_s[t + 1] - P_pred_next) @ J.T return x_s, P_s def kalman_series(y, model, q, r): """Return (filtered_level, smoothed_level). model ∈ {'LL','LLT'}.""" if model == "LL": F, H, Q, R =
